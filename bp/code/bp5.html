Critical Bulkification

Always write code to handle multiple records at once. Salesforce processes records in batches of 200.

‚ùå Bad Practice
Apex Trigger 
trigger AccountTrigger on Account (after insert) {
    for(Account a : Trigger.new) {
        // SOQL in loop!
        Contact c = [SELECT Id FROM Contact WHERE AccountId = :a.Id LIMIT 1];
        // DML in loop!
        update a;
    }
}

Apex Trigger 
trigger AccountTrigger on Account (after insert) {
    Set<Id> accountIds = new Set<Id>();
    for(Account a : Trigger.new) {
        accountIds.add(a.Id);
    }

    // Single SOQL
    List<Contact> contacts = [SELECT Id, AccountId FROM Contact WHERE AccountId IN :accountIds];

    // Process in bulk...
    List<Account> accountsToUpdate = new List<Account>();
    // ...logic here
    update accountsToUpdate; // Single DML
}
