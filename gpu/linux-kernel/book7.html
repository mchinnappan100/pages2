LLM Inference Systems &
Kernel Optimization

Every technique that makes LLMs fast enough to serve at production scale — KV cache mechanics, Flash Attention's IO-aware design, INT4/FP8 quantisation, PagedAttention, speculative decoding, and tensor parallelism — all explained from first principles with working Python code.

The Inference Problem

Training a 70B LLM takes weeks. Serving it to millions of users in real time is the harder engineering challenge. Each token generation requires reading 140 GB of weights (in FP16) from HBM, performing billions of floating-point operations, and returning a result in under 50ms — repeatedly, for thousands of concurrent users.

This book covers every optimization technique that closes the gap between "it runs" and "it serves."

The Core Insight

LLM inference has two distinct phases with radically different performance profiles: Prefill (compute-bound: process the full prompt in one forward pass) and Decode (memory-bandwidth-bound: generate one token at a time, reading all weights each step). Every inference optimization targets one or both of these phases.

Chapter 1 — KV Cache Deep Dive

KV Cache Memory Arithmetic

During autoregressive generation, attention keys and values from all previous positions must be accessible at each new step. Without caching, we'd recompute them from scratch — O(n²) work per token. With caching, we store K and V tensors for every layer and every position — O(n) memory, O(1) compute per new token.

KV Cache Memory Formula

KV_bytes = 2 × dtype_bytes × n_layers × n_kv_heads × d_head × seq_len × batch
For Llama-2 70B (FP16, 80 layers, 8 KV heads, 128 d_head), batch=1, seq=4096:
= 2 × 2 × 80 × 8 × 128 × 4096 × 1 = 1.34 GB per sequence
At batch=100: 134 GB — more than the model weights!

"""
kv_cache.py — KV Cache implementation from scratch.
Shows exactly what vLLM and HuggingFace do internally.
"""
import torch
import torch.nn.functional as F
from dataclasses import dataclass
from typing import Optional, List, Tuple

@dataclass
class KVCacheConfig:
    n_layers:   int = 80
    n_kv_heads: int = 8      # GQA: fewer KV heads than Q heads
    d_head:     int = 128
    max_seq:    int = 4096
    dtype:      torch.dtype = torch.float16
    device:     str = 'cuda'

class KVCache:
    """Pre-allocated KV cache for a single sequence."""

    def __init__(self, cfg: KVCacheConfig):
        self.cfg = cfg
        self.current_len = 0

        # Pre-allocate full max_seq tensors — avoids allocation during decoding
        # Shape: (n_layers, n_kv_heads, max_seq, d_head)
        shape = (cfg.n_layers, cfg.n_kv_heads, cfg.max_seq, cfg.d_head)
        self.k_cache = torch.zeros(shape, dtype=cfg.dtype, device=cfg.device)
        self.v_cache = torch.zeros(shape, dtype=cfg.dtype, device=cfg.device)

        mem_gb = 2 * self.k_cache.numel() * self.k_cache.element_size() / 1e9
        print(f"KV cache allocated: {mem_gb:.2f} GB "
              f"({cfg.n_layers} layers × {cfg.n_kv_heads} heads × "
              f"{cfg.max_seq} seq × {cfg.d_head} d_head × 2 × fp16)")

    def update(self, layer: int, k_new: torch.Tensor,
               v_new: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Append new K,V to cache and return full K,V tensors for attention.

        k_new, v_new: (n_kv_heads, new_tokens, d_head)
        Returns:      (n_kv_heads, current_total_tokens, d_head)
        """
        new_len = k_new.shape[1]  # number of new tokens (1 during decode)

        if self.current_len + new_len > self.cfg.max_seq:
            raise ValueError(f"KV cache overflow: "
                             f"{self.current_len}+{new_len} > {self.cfg.max_seq}")

        # Write new tokens into pre-allocated slots
        end = self.current_len + new_len
        self.k_cache[layer, :, self.current_len:end] = k_new
        self.v_cache[layer, :, self.current_len:end] = v_new

        if new_len == 1:  # decode phase: advance pointer
            self.current_len += 1

        # Return all K,V tokens up to current position
        k_full = self.k_cache[layer, :, :end]
        v_full = self.v_cache[layer, :, :end]
        return k_full, v_full

    def prefill(self, layer: int, k: torch.Tensor,
                v: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """Process the full prompt (prefill phase) and populate cache."""
        seq_len = k.shape[1]
        self.k_cache[layer, :, :seq_len] = k
        self.v_cache[layer, :, :seq_len] = v
        self.current_len = seq_len
        return self.k_cache[layer, :, :seq_len], self.v_cache[layer, :, :seq_len]

    def memory_used_gb(self) -> float:
        """Current memory actually occupied by valid tokens."""
        cfg = self.cfg
        return (2 * cfg.n_layers * cfg.n_kv_heads *
                self.current_len * cfg.d_head *
                2  # fp16 = 2 bytes
                ) / 1e9


def decode_step_with_cache(
    layer_weights,        # dict: W_Q, W_K, W_V, W_O
    x: torch.Tensor,      # (1, 1, d_model) — one new token
    kv_cache: KVCache,
    layer_idx: int,
) -> torch.Tensor:
    """One decode step with KV cache — the hot path in LLM serving."""
    n_heads    = 32    # Q heads (may differ from KV heads with GQA)
    n_kv_heads = kv_cache.cfg.n_kv_heads
    d_head     = kv_cache.cfg.d_head
    scale      = d_head ** -0.5

    # Project to Q, K, V
    q = F.linear(x, layer_weights['W_Q'])  # (1, 1, n_heads*d_head)
    k = F.linear(x, layer_weights['W_K'])  # (1, 1, n_kv_heads*d_head)
    v = F.linear(x, layer_weights['W_V'])  # (1, 1, n_kv_heads*d_head)

    # Reshape to (n_heads, 1, d_head) — squeeze batch dim for clarity
    q = q.squeeze(0).reshape(n_heads, 1, d_head)
    k = k.squeeze(0).reshape(n_kv_heads, 1, d_head)
    v = v.squeeze(0).reshape(n_kv_heads, 1, d_head)

    # Update cache and get full K,V history
    k_full, v_full = kv_cache.update(layer_idx, k, v)
    # k_full: (n_kv_heads, current_len, d_head)

    # GQA: expand KV heads to match Q heads
    groups = n_heads // n_kv_heads
    k_full = k_full.repeat_interleave(groups, dim=0)
    v_full = v_full.repeat_interleave(groups, dim=0)

    # Compute attention: Q is (n_heads,1,d_k), K is (n_heads,T,d_k)
    scores = torch.bmm(q, k_full.transpose(1, 2)) * scale
    # No causal mask needed: new token attends to ALL previous tokens by design
    attn = F.softmax(scores, dim=-1)        # (n_heads, 1, T)
    out  = torch.bmm(attn, v_full)           # (n_heads, 1, d_head)

    # Concatenate heads and project
    out = out.reshape(1, 1, n_heads * d_head)
    return F.linear(out, layer_weights['W_O'])


if __name__ == '__main__':
    cfg = KVCacheConfig(n_layers=80, n_kv_heads=8, d_head=128, max_seq=4096)
    cache = KVCache(cfg)
    print(f"After prefill of 1024 tokens: {cache.memory_used_gb():.3f} GB")

Chapter 2 — PagedAttention & vLLM

OS-Inspired KV Cache Management

Traditional KV caching pre-allocates a contiguous buffer for each request's maximum sequence length — wasting up to 60% of GPU memory on unused slots. vLLM's PagedAttention borrows the OS virtual memory model: store KV cache in fixed-size blocks (like pages), track them with a block table (like a page table), allocate on demand.

"""
paged_attention.py — PagedAttention concept implementation.
Demonstrates the block table mechanism that vLLM uses.
"""
import torch
from typing import Dict, List, Optional

BLOCK_SIZE = 16      # tokens per block (vLLM default)
MAX_BLOCKS = 4096    # total GPU memory pool
N_LAYERS   = 32
N_KV_HEADS = 8
D_HEAD     = 128

class BlockAllocator:
    """GPU memory pool for KV cache blocks — like a physical page allocator."""

    def __init__(self):
        # Physical block storage: (MAX_BLOCKS, N_LAYERS, 2, N_KV_HEADS, BLOCK_SIZE, D_HEAD)
        # 2 = K and V
        self.storage = torch.zeros(
            MAX_BLOCKS, N_LAYERS, 2, N_KV_HEADS, BLOCK_SIZE, D_HEAD,
            dtype=torch.float16, device='cuda'
        )
        self.free_blocks: List[int] = list(range(MAX_BLOCKS))
        self.ref_counts: Dict[int, int] = {}

    def allocate(self) -> int:
        if not self.free_blocks:
            raise RuntimeError("OOM: no free KV cache blocks")
        block_id = self.free_blocks.pop()
        self.ref_counts[block_id] = 1
        return block_id

    def free(self, block_id: int):
        self.ref_counts[block_id] -= 1
        if self.ref_counts[block_id] == 0:
            self.free_blocks.append(block_id)
            del self.ref_counts[block_id]

    def copy_on_write(self, block_id: int) -> int:
        """Fork a shared block — used in beam search / parallel sampling."""
        new_id = self.allocate()
        self.storage[new_id] = self.storage[block_id].clone()
        self.free(block_id)
        return new_id


class SequenceKVCache:
    """Per-sequence KV cache using block table (virtual → physical mapping)."""

    def __init__(self, allocator: BlockAllocator):
        self.allocator  = allocator
        self.block_table: List[int] = []  # virtual block N → physical block ID
        self.current_len = 0

    def _ensure_capacity(self, target_len: int):
        """Allocate new blocks as the sequence grows."""
        needed_blocks = (target_len + BLOCK_SIZE - 1) // BLOCK_SIZE
        while len(self.block_table) < needed_blocks:
            self.block_table.append(self.allocator.allocate())

    def write_kv(self, layer: int, pos: int,
                  k: torch.Tensor, v: torch.Tensor):
        """Write K,V for a single position."""
        self._ensure_capacity(pos + 1)
        block_idx   = pos // BLOCK_SIZE
        slot_in_blk = pos %  BLOCK_SIZE
        phys_block  = self.block_table[block_idx]
        self.allocator.storage[phys_block, layer, 0, :, slot_in_blk, :] = k
        self.allocator.storage[phys_block, layer, 1, :, slot_in_blk, :] = v
        self.current_len = max(self.current_len, pos + 1)

    def get_kv(self, layer: int) -> Tuple[torch.Tensor, torch.Tensor]:
        """Gather all K,V tokens across physical blocks."""
        ks, vs = [], []
        for i in range(self.current_len):
            block_idx  = i // BLOCK_SIZE
            slot       = i %  BLOCK_SIZE
            phys       = self.block_table[block_idx]
            ks.append(self.allocator.storage[phys, layer, 0, :, slot, :])
            vs.append(self.allocator.storage[phys, layer, 1, :, slot, :])
        return torch.stack(ks, dim=1), torch.stack(vs, dim=1)
        # Returns (n_kv_heads, current_len, d_head)

    def free(self):
        for block_id in self.block_table:
            self.allocator.free(block_id)
        self.block_table.clear()

Continuous Batching

Traditional static batching waits for all sequences to finish before starting new ones — wasting GPU when short sequences complete early. Continuous batching (also called iteration-level scheduling) replaces finished sequences immediately, keeping the GPU fully occupied.

"""Continuous batching scheduler — the key insight behind vLLM's throughput."""

from collections import deque
import time

class ContinuousBatchScheduler:
    def __init__(self, max_batch_size: int = 32):
        self.max_batch  = max_batch_size
        self.waiting    = deque()     # requests not yet started
        self.running    = {}          # seq_id → sequence state
        self.completed  = []

    def schedule(self) -> List[dict]:
        """Called before each decode step. Fill batch with new requests."""
        # Remove completed sequences, free their KV cache blocks
        done = [sid for sid, seq in self.running.items() if seq['done']]
        for sid in done:
            seq = self.running.pop(sid)
            seq['kv_cache'].free()     # immediately returns blocks to pool!
            self.completed.append(seq)

        # Fill empty slots from waiting queue — no delay, no idle GPU
        while len(self.running) < self.max_batch and self.waiting:
            new_req = self.waiting.popleft()
            self.running[new_req['id']] = new_req
            print(f"[Scheduler] Started seq {new_req['id']} "
                  f"(batch now {len(self.running)}/{self.max_batch})")

        return list(self.running.values())

Chapter 3 — Flash Attention

IO-Aware Algorithm Design

Standard attention materialises the full T×T score matrix in HBM — O(T²) memory. For T=32K, that's 4 GB just for scores. Flash Attention (Dao et al. 2022) computes exact attention in O(T) memory by processing in tiles that fit in SRAM, fusing the softmax+matmul into one pass.

MetricStandard AttentionFlash Attention 2
HBM memoryO(T²)O(T)
HBM readsO(T² × d)O(T × d)
SRAM usageO(d)O(T_block × d)
Numerical resultExactExact (same math)
Speed (A100, T=4K)1×~4×

Flash Attention in PyTorch (torch.nn.functional)

"""
flash_attention_demo.py — Using Flash Attention in PyTorch.
torch.nn.functional.scaled_dot_product_attention() automatically dispatches
to Flash Attention 2 when available (PyTorch ≥ 2.0, GPU ≥ Ampere).
"""
import torch
import torch.nn.functional as F
import time

def benchmark_attention(
    batch: int = 4,
    heads: int = 32,
    seq:   int = 4096,
    d_k:   int = 128,
    causal: bool = True,
    n_warmup: int = 10,
    n_bench:  int = 100,
):
    device = 'cuda'
    dtype  = torch.float16

    Q = torch.randn(batch, heads, seq, d_k, device=device, dtype=dtype)
    K = torch.randn(batch, heads, seq, d_k, device=device, dtype=dtype)
    V = torch.randn(batch, heads, seq, d_k, device=device, dtype=dtype)

    # ── Standard attention (naive) ──
    def standard_attn():
        scale  = d_k ** -0.5
        scores = torch.matmul(Q, K.transpose(-2, -1)) * scale
        if causal:
            mask = torch.triu(torch.full((seq, seq), float('-inf'),
                              device=device), diagonal=1)
            scores = scores + mask
        return torch.matmul(F.softmax(scores, dim=-1), V)

    # ── Flash Attention (via SDPA dispatch) ──
    def flash_attn():
        with torch.backends.cuda.sdp_kernel(
            enable_flash=True, enable_math=False, enable_mem_efficient=False
        ):
            return F.scaled_dot_product_attention(Q, K, V, is_causal=causal)

    # Verify numerical equivalence
    out_std   = standard_attn()
    out_flash = flash_attn()
    max_err = (out_std - out_flash).abs().max().item()
    print(f"Max numerical error (std vs flash): {max_err:.6f}")

    # Benchmark
    for name, fn in [("Standard", standard_attn), ("Flash  ", flash_attn)]:
        for _ in range(n_warmup):
            fn()
        torch.cuda.synchronize()
        t0 = time.perf_counter()
        for _ in range(n_bench):
            fn()
        torch.cuda.synchronize()
        t1 = time.perf_counter()
        ms_per = (t1 - t0) / n_bench * 1000
        # Memory estimate for T×T score matrix
        score_mem_gb = batch * heads * seq * seq * 2 / 1e9
        print(f"{name}: {ms_per:.2f} ms/iter "
              f"(score matrix would be {score_mem_gb:.2f} GB)")

if __name__ == '__main__':
    print(f"PyTorch {torch.__version__}, "
          f"Flash Attention available: {torch.backends.cuda.flash_sdp_enabled()}")
    benchmark_attention(batch=4, heads=32, seq=4096, d_k=128)

Chapter 4 — Quantisation: INT4 & FP8

Why Quantisation Works

LLM weights are over-parameterised: small perturbations in weight values have negligible effect on outputs. INT4 represents each weight with only 4 bits (16 possible values) instead of 16 bits — 4× less memory, 4× faster to read from HBM. The key challenge is calibrating the quantisation range to minimise the accuracy drop.

Implementing INT8 Weight Quantisation

"""
quantization.py — Manual INT8 weight quantisation from scratch.
Shows the per-channel quantisation that bitsandbytes and GPTQ use internally.
"""
import torch
import torch.nn as nn

class Int8Linear(nn.Module):
    """A linear layer with INT8 weights and FP16 activations."""

    def __init__(self, weight: torch.Tensor):
        super().__init__()
        # Quantise weights to INT8 at construction time
        self.w_quant, self.scale = self._quantize_weight(weight)
        self.out_features, self.in_features = weight.shape

    @staticmethod
    def _quantize_weight(weight: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """Per-channel symmetric INT8 quantisation.
        Per-channel = separate scale per output neuron (row of weight matrix).
        This is critical: per-tensor quantisation would lose too much accuracy."""
        # Find max absolute value per output channel (row)
        scale = weight.abs().max(dim=1).values / 127.0  # (out_features,)
        scale = scale.clamp(min=1e-8)                    # avoid div by zero

        # Quantise: divide by scale, round, clamp to [-127, 127]
        w_quant = (weight / scale[:, None]).round().clamp(-127, 127).to(torch.int8)

        # Measure quantisation error
        w_dequant = w_quant.to(torch.float32) * scale[:, None]
        error = (weight - w_dequant).abs().mean().item()
        print(f"INT8 quantisation error: {error:.6f} "
              f"(weight shape {tuple(weight.shape)})")

        return w_quant, scale.to(torch.float16)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Dequantise on the fly (INT8 → FP16) then compute matmul
        # In practice, INT8 GEMM kernels (e.g., cuBLASLt) avoid full dequantisation
        w_fp16 = self.w_quant.to(torch.float16) * self.scale[:, None]
        return F.linear(x, w_fp16)

    def memory_footprint(self) -> dict:
        int8_bytes  = self.w_quant.numel()                  # 1 byte each
        scale_bytes = self.scale.numel() * 2               # fp16 = 2 bytes
        fp16_bytes  = self.in_features * self.out_features * 2  # baseline
        return {
            'fp16_MB':  fp16_bytes / 1e6,
            'int8_MB':  (int8_bytes + scale_bytes) / 1e6,
            'ratio':    fp16_bytes / (int8_bytes + scale_bytes),
        }


def quantize_model_int8(model: nn.Module) -> nn.Module:
    """Replace all Linear layers with Int8Linear equivalents."""
    for name, module in model.named_modules():
        if isinstance(module, nn.Linear):
            parent_name, child_name = name.rsplit('.', 1) if '.' in name else ('', name)
            parent = model.get_submodule(parent_name) if parent_name else model
            q_layer = Int8Linear(module.weight.data.to(torch.float32))
            setattr(parent, child_name, q_layer)
            print(f"  Quantised {name}: {q_layer.memory_footprint()}")
    return model


if __name__ == '__main__':
    # Simulate quantising a transformer FFN layer (d_model=4096, d_ff=16384)
    ffn = nn.Sequential(nn.Linear(4096, 16384), nn.GELU(), nn.Linear(16384, 4096))
    print("Quantising FFN layers to INT8...")
    ffn_q = quantize_model_int8(ffn)

    # Verify output is similar
    x = torch.randn(1, 32, 4096)
    y_fp16 = ffn(x)
    y_int8 = ffn_q(x)
    err = (y_fp16 - y_int8).abs().mean().item()
    print(f"FFN output error (fp16 vs int8): {err:.6f}")

Chapter 5 — Speculative Decoding

"""
speculative_decoding.py — Speed up LLM inference 2-3× with a small draft model.

Algorithm:
1. Small draft model generates K tokens quickly
2. Large target model verifies all K tokens in ONE forward pass (parallel!)
3. Accept tokens up to first mismatch; resample the rest
4. Expected speedup: K × acceptance_rate tokens per target model call
"""
import torch
import torch.nn.functional as F
from typing import List, Tuple

def speculative_decode(
    draft_model,        # small fast model (e.g., 7B)
    target_model,       # large accurate model (e.g., 70B)
    input_ids: torch.Tensor,
    n_draft: int = 4,  # number of draft tokens to generate
    max_new_tokens: int = 128,
    temperature: float = 1.0,
) -> torch.Tensor:
    """
    Speculative decoding loop.
    Returns extended input_ids with generated tokens appended.
    """
    generated = input_ids.clone()
    n_accepted_total = 0
    n_target_calls = 0

    while generated.shape[1] - input_ids.shape[1] < max_new_tokens:

        # ── Phase 1: Draft model generates K tokens ──
        draft_tokens  = []
        draft_probs   = []
        ctx = generated.clone()

        with torch.no_grad():
            for _ in range(n_draft):
                logits = draft_model(ctx).logits[:, -1, :]  # (B, vocab)
                probs  = F.softmax(logits / temperature, dim=-1)
                token  = torch.multinomial(probs, num_samples=1)
                draft_tokens.append(token)
                draft_probs.append(probs)
                ctx = torch.cat([ctx, token], dim=1)

        # ── Phase 2: Target model verifies ALL draft tokens in ONE call ──
        candidate = torch.cat([generated] + draft_tokens, dim=1)
        with torch.no_grad():
            target_logits = target_model(candidate).logits   # (B, L+K, vocab)
        n_target_calls += 1

        # Target probabilities at each draft position
        target_probs = F.softmax(target_logits / temperature, dim=-1)

        # ── Phase 3: Accept/reject via speculative sampling ──
        n_accepted = 0
        for i in range(n_draft):
            pos         = generated.shape[1] + i - 1
            token       = draft_tokens[i]
            p_draft     = draft_probs[i][0, token]      # prob under draft model
            p_target    = target_probs[0, pos, token]   # prob under target model

            # Accept with probability min(1, p_target / p_draft)
            accept_prob = torch.min(torch.tensor(1.0), p_target / p_draft)
            if torch.rand(1) < accept_prob:
                n_accepted += 1
            else:
                break   # reject this and all subsequent tokens

        # Append accepted tokens
        if n_accepted > 0:
            generated = torch.cat([generated] + draft_tokens[:n_accepted], dim=1)

        # Always append one target-sampled token (even if 0 accepted)
        bonus_pos   = generated.shape[1] - 1
        bonus_probs = target_probs[0, bonus_pos]
        if n_accepted < n_draft:
            # Resample from adjusted distribution
            adj_probs = (target_probs[0, bonus_pos] - draft_probs[n_accepted][0]).clamp(min=0)
            adj_probs = adj_probs / adj_probs.sum()
            bonus_token = torch.multinomial(adj_probs, num_samples=1)
        else:
            bonus_token = torch.multinomial(bonus_probs, num_samples=1)
        generated = torch.cat([generated, bonus_token[None]], dim=1)

        n_accepted_total += n_accepted + 1  # +1 for bonus token

    tokens_per_call = n_accepted_total / n_target_calls
    print(f"Avg tokens per target call: {tokens_per_call:.2f} "
          f"(theoretical max: {n_draft + 1})")
    return generated

Chapter 6 — Tensor Parallelism for Inference

"""
tensor_parallel.py — Column and row parallel linear layers.
This is how Megatron-LM splits transformer layers across GPUs.
"""
import torch
import torch.nn as nn
import torch.distributed as dist

class ColumnParallelLinear(nn.Module):
    """Split weight matrix along output (column) dimension.
    Each GPU holds 1/TP of the output features.
    Input is broadcast; output is gathered with AllReduce."""

    def __init__(self, in_features: int, out_features: int, tp_size: int):
        super().__init__()
        self.tp_size = tp_size
        self.rank    = dist.get_rank() if dist.is_initialized() else 0
        local_out = out_features // tp_size
        self.weight = nn.Parameter(torch.empty(local_out, in_features))
        nn.init.kaiming_uniform_(self.weight)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Each GPU computes partial output with its shard of weight
        out = torch.nn.functional.linear(x, self.weight)  # (B, T, local_out)
        # AllGather to collect all shards: (B, T, total_out)
        if self.tp_size > 1 and dist.is_initialized():
            gathered = [torch.zeros_like(out) for _ in range(self.tp_size)]
            dist.all_gather(gathered, out)
            out = torch.cat(gathered, dim=-1)
        return out


class RowParallelLinear(nn.Module):
    """Split weight matrix along input (row) dimension.
    Input is split; partial outputs are summed with AllReduce."""

    def __init__(self, in_features: int, out_features: int, tp_size: int):
        super().__init__()
        self.tp_size = tp_size
        self.rank    = dist.get_rank() if dist.is_initialized() else 0
        local_in = in_features // tp_size
        self.weight = nn.Parameter(torch.empty(out_features, local_in))
        nn.init.kaiming_uniform_(self.weight)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x is already split along last dim (came from ColumnParallel)
        out = torch.nn.functional.linear(x, self.weight)  # (B, T, out)
        # AllReduce: sum partial results across TP group
        if self.tp_size > 1 and dist.is_initialized():
            dist.all_reduce(out, op=dist.ReduceOp.SUM)
        return out


# Tensor-parallel FFN: split intermediate dimension across TP_SIZE GPUs
# GPU 0: handles first 4096 neurons of the 16384-wide hidden layer
# GPU 1: handles next 4096, etc.
# AllReduce at the end synchronises the result.
class TensorParallelFFN(nn.Module):
    def __init__(self, d_model: int, d_ff: int, tp_size: int):
        super().__init__()
        self.fc1 = ColumnParallelLinear(d_model, d_ff, tp_size)
        self.fc2 = RowParallelLinear(d_ff, d_model, tp_size)

    def forward(self, x):
        return self.fc2(torch.nn.functional.gelu(self.fc1(x)))
