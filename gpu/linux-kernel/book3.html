Chapter 1 — VFS: The Virtual Filesystem

The VFS Abstraction Layers

Linux's Virtual Filesystem (VFS) provides a unified interface to all filesystem types — ext4, XFS, tmpfs, NFS, and more. When your data loader calls open(), VFS routes the request to the correct filesystem driver transparently.

InodeinodeFile metadata (size, permissions, timestamps, block locations)
DentrydentryDirectory cache: maps filename → inode (path lookup cache)
FilefilePer-open-file state: position, flags, open mode

inode and file_operations

/* The file_operations struct: VFS dispatch table for every file I/O call.
 * When you call read() on a file, VFS calls file->f_op->read().
 * Filesystem drivers (ext4, xfs) implement these to do actual disk I/O. */

/* Simplified inode structure */
struct inode {
    umode_t          i_mode;       /* file type and permissions */
    kuid_t           i_uid;
    kgid_t           i_gid;
    loff_t           i_size;       /* file size in bytes */
    blkcnt_t         i_blocks;     /* number of 512-byte blocks */
    struct timespec64 i_mtime;
    const struct inode_operations *i_op;  /* inode ops (rename, link, etc) */
    const struct file_operations  *i_fop; /* file ops (read, write, mmap) */
    struct address_space *i_mapping;      /* page cache for this file */
};

/* How the page cache accelerates repeated reads:
 * First read: disk → page cache (RAM) → user buffer
 * Second read: page cache (RAM) → user buffer  (zero disk I/O!)
 *
 * For ML training: with O_DIRECT, bypass page cache entirely.
 * This prevents training data from evicting model weights from RAM. */
#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

void read_dataset_direct(const char* path) {
    /* O_DIRECT: bypass page cache; requires 512-byte aligned buffers */
    int fd = open(path, O_RDONLY | O_DIRECT | O_NOATIME);
    if (fd < 0) { perror("open"); return; }

    /* Must be aligned to 512 bytes (sector size) for O_DIRECT */
    size_t buf_size = 4 * 1024 * 1024;  /* 4 MB read buffer */
    void*  buf;
    posix_memalign(&buf, 4096, buf_size);  /* 4096-byte aligned */

    ssize_t n;
    size_t  total = 0;
    while ((n = read(fd, buf, buf_size)) > 0)
        total += n;

    printf("Read %zu MB directly from NVMe (no page cache)\n",
           total / (1024 * 1024));
    free(buf); close(fd);
}

Chapter 2 — io_uring: Async I/O for Data Loading

The Ring Buffer Design

io_uring (introduced in kernel 5.1) is the fastest I/O mechanism in Linux. It uses two shared ring buffers between kernel and user space — a Submission Queue (SQ) and a Completion Queue (CQ) — and can process I/O operations with zero system calls in the steady state.

I/O MethodSyscalls per opMax throughputLatency
read()/write()1 per op~800K IOPS~15 µs
preadv2()1 per batch~1.2M IOPS~10 µs
io_uring (polled)~0 (kernel polls)~4M+ IOPS~2 µs

Async Dataset Loading with io_uring

#include <liburing.h>    /* link with -luring */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>

#define QUEUE_DEPTH  256   /* outstanding I/O requests */
#define BATCH_SIZE    32   /* files per batch (one DataLoader batch) */
#define IMG_SIZE  (224*224*3)  /* raw RGB image size */

struct io_request {
    int   fd;
    char  *buf;
    int   index;
};

/* Load BATCH_SIZE images asynchronously using io_uring.
 * All reads are submitted at once, then all completions are collected.
 * This is the pattern for a high-throughput ML data loader. */
int async_load_batch(const char** filenames, char** buffers, int n) {
    struct io_uring ring;
    int ret;

    /* Initialize io_uring with IOSQE_FIXED_FILE and IORING_FEAT_FAST_POLL
     * for maximum NVMe performance */
    ret = io_uring_queue_init(QUEUE_DEPTH, &ring, IORING_SETUP_SQPOLL);
    if (ret < 0) {
        fprintf(stderr, "io_uring_queue_init failed: %s\n", strerror(-ret));
        return -1;
    }

    /* Open all files and submit all read requests */
    int fds[BATCH_SIZE];
    struct io_request reqs[BATCH_SIZE];

    for (int i = 0; i < n; i++) {
        fds[i] = open(filenames[i], O_RDONLY | O_DIRECT);
        if (fds[i] < 0) { perror(filenames[i]); continue; }

        reqs[i].fd    = fds[i];
        reqs[i].buf   = buffers[i];
        reqs[i].index = i;

        /* Get a Submission Queue Entry */
        struct io_uring_sqe *sqe = io_uring_get_sqe(&ring);
        /* Prepare an async read: fd, buffer, size, offset */
        io_uring_prep_read(sqe, fds[i], buffers[i], IMG_SIZE, 0);
        /* Tag the SQE with our request pointer for the completion handler */
        io_uring_sqe_set_data(sqe, &reqs[i]);
    }

    /* Submit ALL reads in one syscall (or zero if SQPOLL kernel thread is active) */
    ret = io_uring_submit(&ring);
    printf("Submitted %d async reads in 1 syscall\n", ret);

    /* Collect completions */
    int completed = 0;
    while (completed < n) {
        struct io_uring_cqe *cqe;
        ret = io_uring_wait_cqe(&ring, &cqe);
        if (ret < 0) break;

        struct io_request *req = io_uring_cqe_get_data(cqe);
        if (cqe->res < 0)
            fprintf(stderr, "Read[%d] failed: %s\n", req->index, strerror(-cqe->res));

        io_uring_cqe_seen(&ring, cqe);  /* advance CQ head */
        close(fds[req->index]);
        completed++;
    }
    printf("Completed %d reads\n", completed);
    io_uring_queue_exit(&ring);
    return completed;
}

Chapter 3 — NVMe Storage for Training Data

NVMe Queue Depth and Parallelism

NVMe SSDs support up to 65,535 I/O queues with 65,535 commands per queue — far more than SATA's single queue with 32 commands. For AI data loading, this parallelism is critical: you want many outstanding I/O requests simultaneously.

#!/bin/bash
# Tune NVMe for maximum AI training data throughput

# Find NVMe devices
lsblk -d -o NAME,TYPE,ROTA | grep -v ROTA=1

# Check current I/O scheduler (should be 'none' for NVMe)
cat /sys/block/nvme0n1/queue/scheduler
# Set to 'none' — NVMe's internal controller is smarter than any kernel scheduler
echo none > /sys/block/nvme0n1/queue/scheduler

# Set queue depth (default 128, max 1024 for NVMe)
echo 1024 > /sys/block/nvme0n1/queue/nr_requests

# Enable write-back caching (if safe for your workload)
hdparm -W1 /dev/nvme0n1

# Benchmark random read IOPS (simulates DataLoader access pattern)
fio --name=rand_read \
    --filename=/dev/nvme0n1 \
    --ioengine=io_uring \
    --direct=1 \
    --rw=randread \
    --bs=128k \         # typical compressed image/audio file size
    --numjobs=32 \      # parallel DataLoader workers
    --iodepth=64 \      # outstanding I/Os per worker
    --runtime=30 \
    --time_based \
    --output-format=terse

# For RAID-0 across 4 NVMe drives (4× bandwidth):
# mdadm --create /dev/md0 --level=0 --raid-devices=4 /dev/nvme[0-3]n1

Python DataLoader Benchmark

"""
dataloader_bench.py — Benchmark PyTorch DataLoader I/O throughput.
Shows how num_workers, pin_memory, and prefetch_factor interact.
"""
import torch, time
from torch.utils.data import Dataset, DataLoader
import numpy as np
from pathlib import Path

class ImageDataset(Dataset):
    def __init__(self, root: str, n_samples: int = 10000):
        self.files = list(Path(root).glob('**/*.jpg'))[:n_samples]

    def __len__(self): return len(self.files)

    def __getitem__(self, idx):
        # Raw file read — tests pure I/O bandwidth
        data = np.frombuffer(self.files[idx].read_bytes(), dtype=np.uint8)
        return torch.from_numpy(data[:224*224*3].reshape(3, 224, 224).copy())

def benchmark(dataset, num_workers: int, pin_memory: bool, prefetch: int):
    loader = DataLoader(
        dataset,
        batch_size=256,
        num_workers=num_workers,
        pin_memory=pin_memory,
        prefetch_factor=prefetch if num_workers > 0 else 2,
        persistent_workers=num_workers > 0,  # keep workers alive between epochs
    )
    # Warmup
    for i, _ in enumerate(loader):
        if i >= 5: break

    t0 = time.perf_counter()
    n_samples = 0
    for batch in loader:
        n_samples += batch.shape[0]
    t1 = time.perf_counter()

    throughput = n_samples / (t1 - t0)
    print(f"workers={num_workers:2d} pin={pin_memory} prefetch={prefetch}: "
          f"{throughput:6.0f} samples/sec")
    return throughput

if __name__ == '__main__':
    ds = ImageDataset('/path/to/imagenet/train')
    print(f"Dataset: {len(ds)} images\n")
    for nw in [0, 4, 8, 16]:
        for pm in [False, True]:
            benchmark(ds, nw, pm, prefetch=4)

Chapter 4 — Linux Network Stack

The Packet Receive Path

When a NCCL gradient arrives from another server, it traverses the Linux network stack: NIC DMA → kernel RX ring → softirq processing → socket buffer → user space. Understanding this path is essential for diagnosing network bottlenecks in distributed training.

/* The receive-side scaling (RSS) and interrupt affinity setup
 * that NCCL documentation recommends but rarely explains WHY. */
/*
 * Packet receive path (simplified):
 * 1. NIC receives packet, DMA to kernel ring buffer (no CPU involvement)
 * 2. NIC raises hardware interrupt (IRQ)
 * 3. IRQ handler runs on one CPU: acks interrupt, schedules NAPI poll
 * 4. NAPI softirq runs: drains ring buffer, builds sk_buff, passes to IP stack
 * 5. TCP/IP processing: sequence checking, ACK generation
 * 6. Data copied to socket receive buffer
 * 7. Waiting process woken up (e.g., NCCL recv thread)
 *
 * Optimization: pin NIC IRQs to CPUs local to the GPU doing the AllReduce.
 * This reduces cross-socket latency in the packet path.
 */

Zero-Copy for Checkpoint Transfer

#include <sys/sendfile.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <fcntl.h>
#include <stdio.h>
#include <sys/stat.h>

/* sendfile() transfers data from file to socket entirely in kernel space.
 * NO data copies through user space. Critical for fast checkpoint distribution.
 *
 * Without sendfile:  file → kernel buffer → USER BUFFER → kernel socket → NIC
 * With sendfile:     file → kernel socket → NIC  (zero user-space copies!)
 */
int distribute_checkpoint(const char* ckpt_path, int dst_sockfd) {
    int         src_fd = open(ckpt_path, O_RDONLY);
    struct stat st;
    off_t       offset = 0;

    if (src_fd < 0) { perror("open"); return -1; }
    fstat(src_fd, &st);

    printf("Sending checkpoint: %.1f GB\n",
           (double)st.st_size / (1024*1024*1024));

    ssize_t sent = sendfile(dst_sockfd, src_fd,
                             &offset,   /* current file offset (updated in-place) */
                             st.st_size);
    printf("Sent %zd bytes (%.1f GB)\n", sent,
           (double)sent / (1024*1024*1024));

    close(src_fd);
    return (sent == st.st_size) ? 0 : -1;
}

Chapter 5 — RDMA & InfiniBand

Kernel Bypass with RDMA

RDMA (Remote Direct Memory Access) bypasses the kernel network stack entirely. The NIC reads from or writes to remote memory directly, controlled by user-space code. This is how NCCL achieves sub-microsecond latency for AllReduce.

"""
rdma_check.py — Verify RDMA connectivity between nodes.
Run on both nodes simultaneously to test InfiniBand bandwidth.
"""
import subprocess, os

def check_ib_devices():
    """List available InfiniBand devices and their state."""
    result = subprocess.run(['ibv_devices'], capture_output=True, text=True)
    print("InfiniBand devices:")
    print(result.stdout)

    # Show port state for each device
    for dev in ['mlx5_0', 'mlx5_1', 'mlx5_2', 'mlx5_3']:
        result = subprocess.run(['ibv_devinfo', '-d', dev],
                                  capture_output=True, text=True)
        if result.returncode == 0:
            for line in result.stdout.split('\n'):
                if 'state' in line.lower() or 'rate' in line.lower():
                    print(f"  {dev}: {line.strip()}")

def benchmark_ib_bandwidth(server_ip: str):
    """Run ib_send_bw benchmark — measures raw IB bandwidth."""
    # On the server: ib_send_bw
    # On the client: ib_send_bw 
    cmd = ['ib_send_bw', server_ip,
           '-d', 'mlx5_0',
           '-q', '1',    # queue pairs
           '-s', '65536',  # message size (64KB)
           '-n', '1000']   # iterations
    print(f"Running: {' '.join(cmd)}")
    subprocess.run(cmd)

if __name__ == '__main__':
    check_ib_devices()

Chapter 6 — Network Tuning for NCCL

#!/bin/bash
# network_tuning.sh — Tune Linux network stack for NCCL AllReduce performance

## 1. Increase socket buffers (default 212 KB is too small for 400 Gb/s IB) ##
sysctl -w net.core.rmem_max=134217728      # 128 MB RX buffer
sysctl -w net.core.wmem_max=134217728
sysctl -w net.ipv4.tcp_rmem="4096 87380 134217728"
sysctl -w net.ipv4.tcp_wmem="4096 65536 134217728"

## 2. Increase NIC ring buffer sizes ##
ethtool -G mlx5p0 rx 8192 tx 8192

## 3. Enable interrupt coalescing — batch interrupts for efficiency ##
ethtool -C mlx5p0 rx-usecs 50 tx-usecs 50

## 4. Set MTU to jumbo frames (9000 bytes) for lower CPU overhead ##
ip link set dev mlx5p0 mtu 9000

## 5. Disable TCP Nagle (critical for latency-sensitive NCCL) ##
sysctl -w net.ipv4.tcp_nodelay=1

## 6. NCCL-specific: test AllReduce bandwidth ##
# (assumes nccl-tests is built: https://github.com/NVIDIA/nccl-tests)
mpirun -np 8 ./nccl-tests/build/all_reduce_perf \
    -b 1G -e 8G -f 2 \
    -g 1 \
    --nthreads 1

echo "Network tuning complete"
echo "Expected NCCL AllReduce: ~350 GB/s on 8×H100 + NDR InfiniBand"
bash — network_tuning.sh
NCCL Debugging Checklist

If NCCL AllReduce bandwidth is below 50% of theoretical peak: ① Check MTU is 9000 on all IB interfaces. ② Verify NCCL_IB_GID_INDEX matches your IB configuration (ibv_devinfo -d mlx5_0 | grep GID). ③ Run ibstat mlx5_0 and confirm all ports are ACTIVE, not INIT. ④ Check for PCIe congestion: nvidia-smi topo -m should show NVLink connections, not PCIe, between GPUs on the same node.
