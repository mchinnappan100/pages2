Monolithic Architecture — Why It Wins for AI

Linux is a monolithic kernel — all core services run in the same kernel address space. This is a deliberate performance choice: a system call from PyTorch's tensor.cuda() does not cross a process boundary to reach the GPU driver. The latency is microseconds, not milliseconds.

This architectural choice pays enormous dividends in LLM inference where every microsecond of scheduling latency affects tail latency SLAs.

Chapter 2 — Kernel Space vs User Space

CPU Protection Rings

x86-64 CPUs implement four privilege levels called rings. Linux uses only two: Ring 0 (kernel mode) and Ring 3 (user mode). When your Python process calls torch.cuda.synchronize(), the CPU transitions from Ring 3 to Ring 0 via a special syscall instruction.

/*
 * Ring 0: Kernel mode
 *   - Can execute privileged instructions (HLT, LGDT, LIDT...)
 *   - Can access all physical memory
 *   - GPU driver runs here — direct PCIe BAR register access
 *
 * Ring 3: User mode
 *   - Cannot access hardware registers directly
 *   - Cannot access other processes' memory
 *   - PyTorch, CUDA runtime, your training loop runs here
 */

/* A system call from user space looks like this in assembly */
/* mov rax, __NR_ioctl  ; system call number */
/* mov rdi, gpu_fd      ; file descriptor for /dev/nvidia0 */
/* syscall              ; jump to ring 0 */

The System Call Interface — The Bridge PyTorch Uses

Every time PyTorch allocates GPU memory, launches a kernel, or synchronises a stream, it ultimately calls into the kernel via the system call interface. The most important syscalls for AI workloads:

mmap() ioctl() mlock() clone() futex() read()/write() io_uring_enter() 
#include <unistd.h>
#include <sys/mman.h>
#include <stdio.h>
#include <stdint.h>

/* Demonstrate what PyTorch's pinned memory allocation does under the hood.
 * Pinned (page-locked) memory enables DMA transfers without CPU involvement,
 * critical for fast host→GPU tensor copies during data loading. */

void* alloc_pinned(size_t size) {
    void* ptr = mmap(
        NULL, size,
        PROT_READ | PROT_WRITE,
        MAP_PRIVATE | MAP_ANONYMOUS,
        -1, 0
    );
    if (ptr == MAP_FAILED) {
        perror("mmap failed");
        return NULL;
    }
    /* mlock() prevents the OS from swapping these pages to disk.
     * This is REQUIRED for DMA — the physical address must remain stable. */
    if (mlock(ptr, size) != 0) {
        perror("mlock failed — need CAP_IPC_LOCK or increase RLIMIT_MEMLOCK");
        munmap(ptr, size);
        return NULL;
    }
    printf("Pinned %zu bytes at %p\n", size, ptr);
    return ptr;
}

int main() {
    /* Allocate 1 GB of pinned memory — typical for a data-loader buffer */
    size_t size = 1UL * 1024 * 1024 * 1024;
    void* buf = alloc_pinned(size);
    if (!buf) return 1;
    printf("Success! This buffer can now be used for zero-copy GPU DMA.\n");
    munlock(buf, size);
    munmap(buf, size);
    return 0;
}

AI Training Tip

PyTorch's DataLoader(pin_memory=True) calls exactly this sequence. On machines with >1 TB RAM, you may need to raise RLIMIT_MEMLOCK in /etc/security/limits.conf or grant your training process CAP_IPC_LOCK. Forgetting this is a common cause of slow data loading on large GPU clusters.

Chapter 3 — Processes and Threads

The task_struct — The Heart of Every Process

Every process (and thread) in Linux is represented by a task_struct — a C structure defined in include/linux/sched.h that is over 700 fields long. This single structure holds everything the kernel knows about a running entity.

/* Simplified task_struct — showing fields most relevant to AI workloads.
 * Full definition: linux/include/linux/sched.h */
struct task_struct {
    /* ── Scheduling state ── */
    volatile long   state;          /* TASK_RUNNING, TASK_INTERRUPTIBLE, etc */
    int             prio;           /* current priority */
    int             static_prio;    /* nice-based static priority */
    struct sched_entity se;         /* CFS scheduling entity (vruntime lives here) */

    /* ── Memory management ── */
    struct mm_struct *mm;           /* NULL for kernel threads */
    struct mm_struct *active_mm;    /* current active address space */

    /* ── Process identity ── */
    pid_t           pid;            /* process ID */
    pid_t           tgid;           /* thread group ID (= pid of main thread) */
    struct task_struct *parent;     /* pointer to parent process */
    struct list_head children;      /* list of child processes */

    /* ── CPU affinity (CRITICAL for NUMA-aware AI training) ── */
    cpumask_t       cpus_allowed;   /* which CPUs this task can run on */
    int             numa_preferred_nid; /* preferred NUMA node */

    /* ── File descriptors ── */
    struct files_struct *files;     /* open file table (incl. /dev/nvidia0) */

    /* ── cgroups (resource limits for AI containers) ── */
    struct css_set  *cgroups;       /* cgroup membership for memory/CPU limits */

    /* ── Signal handling ── */
    sigset_t        blocked;
    struct sigpending pending;

    /* ... 690+ more fields ... */
};

fork(), exec(), clone() — How Training Workers Are Created

When PyTorch's multiprocessing.spawn() creates worker processes for data loading, it calls clone() — a more flexible version of fork() that lets you specify exactly which resources are shared between parent and child.

#define _GNU_SOURCE
#include <sched.h>
#include <stdlib.h>
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

#define STACK_SIZE (1024 * 1024)   /* 1 MB stack for child */

int worker_fn(void *arg) {
    int worker_id = *(int*)arg;
    printf("[Worker %d] PID=%d starting data-loading loop\n",
           worker_id, getpid());
    /* In real code: open dataset files, prefetch batches into shared memory */
    sleep(1);
    return 0;
}

int main() {
    int num_workers = 4;     /* DataLoader(num_workers=4) */

    for (int i = 0; i < num_workers; i++) {
        char *stack = malloc(STACK_SIZE);
        int  *id    = malloc(sizeof(int));
        *id = i;

        /* CLONE_VM:      share virtual memory (fast, no page table copy)
         * CLONE_FS:      share filesystem root
         * CLONE_FILES:   share file descriptor table
         * SIGCHLD:       notify parent on exit */
        pid_t pid = clone(worker_fn,
                         stack + STACK_SIZE,       /* stack grows downward */
                         CLONE_VM | CLONE_FS | CLONE_FILES | SIGCHLD,
                         id);
        if (pid < 0) { perror("clone"); return 1; }
        printf("[Main] Spawned worker %d with PID %d\n", i, pid);
    }

    /* Wait for all workers */
    for (int i = 0; i < num_workers; i++) wait(NULL);
    printf("All workers done.\n");
    return 0;
}

Chapter 4 — The CFS Scheduler

Completely Fair Scheduler — vruntime and the Red-Black Tree

The Completely Fair Scheduler (CFS) is the default Linux process scheduler since kernel 2.6.23. Its core idea is simple but elegant: track how much CPU time each runnable task has received (its virtual runtime, stored as vruntime), and always run the task with the smallest vruntime next.

CFS maintains all runnable tasks in a red-black tree ordered by vruntime. Selecting the next task is O(log n). The leftmost node (smallest vruntime) is always cached for O(1) access.

/* The CFS scheduling entity — embedded in task_struct as 'se' */
struct sched_entity {
    struct load_weight  load;       /* weight based on nice value */
    struct rb_node      run_node;   /* node in the CFS red-black tree */
    u64                 vruntime;   /* virtual runtime — KEY FIELD */
    u64                 exec_start; /* time task started current run */
    u64                 sum_exec_runtime; /* total CPU time consumed */
};

/* vruntime is updated on every context switch:
 *
 *   delta_exec = now - exec_start
 *   delta_vruntime = delta_exec * (NICE_0_LOAD / task->weight)
 *   task->vruntime += delta_vruntime
 *
 * A nice-0 task accumulates vruntime at the same rate as wall-clock time.
 * A nice-20 task (lower priority) accumulates vruntime FASTER,
 * so it gets scheduled less often. */

/* For AI training: your GPU-intensive processes should have nice=-5 or
 * use SCHED_BATCH to prevent interactive tasks from stealing CPU. */

Setting Scheduler Priority for AI Processes

#include <sys/resource.h>
#include <sched.h>
#include <stdio.h>
#include <unistd.h>

void tune_scheduler_for_training() {
    /* Lower nice = higher priority. Range: -20 (highest) to +19 (lowest).
     * Most training jobs benefit from nice=-5 to nice=-10. */
    if (setpriority(PRIO_PROCESS, 0, -10) != 0) {
        perror("setpriority (needs CAP_SYS_NICE)");
    }

    /* Alternatively, use SCHED_BATCH for throughput-oriented workloads.
     * SCHED_BATCH: assume the task is CPU-bound, wake up less aggressively. */
    struct sched_param param = { .sched_priority = 0 };
    if (sched_setscheduler(0, SCHED_BATCH, &param) != 0) {
        perror("sched_setscheduler");
    }

    printf("Scheduler tuned for batch AI training workload.\n");
}

NUMA-Aware Scheduling for Multi-Socket GPU Servers

Modern GPU servers (e.g., DGX H100) have multiple CPU sockets, each with local DRAM and local PCIe slots (and thus local GPUs). The kernel's NUMA-aware scheduler tracks which NUMA node each task most recently ran on and tries to keep it there — reducing cross-socket memory latency.

/* Pin process to NUMA node 0 and CPU cores 0-31.
 * On a dual-socket H100 server, GPU 0-3 are on NUMA node 0,
 * GPU 4-7 on NUMA node 1. Pin your training process to the right node. */
#define _GNU_SOURCE
#include <numa.h>        /* link with -lnuma */
#include <sched.h>
#include <stdio.h>

void pin_to_numa_node(int node) {
    if (numa_available() < 0) {
        printf("NUMA not available\n"); return;
    }
    printf("System has %d NUMA nodes\n", numa_max_node() + 1);

    /* Run ONLY on CPUs belonging to this NUMA node */
    struct bitmask *cpus = numa_allocate_cpumask();
    numa_node_to_cpus(node, cpus);
    numa_sched_setaffinity(0, cpus);   /* 0 = current process */

    /* Allocate memory ONLY from this NUMA node */
    numa_set_preferred(node);
    numa_set_localalloc();

    printf("Pinned to NUMA node %d. "
           "Memory bandwidth to local GPUs maximised.\n", node);

    numa_free_cpumask(cpus);
}

Chapter 5 — Signals, Futex & IPC

futex — The Primitive Behind PyTorch's Synchronisation

The Fast Userspace muTEX (futex) is the kernel primitive that underlies Python's threading.Lock(), multiprocessing barriers, and distributed training synchronisation. Understanding futex explains why lock contention is so expensive in multi-worker training.

#include <linux/futex.h>
#include <sys/syscall.h>
#include <unistd.h>
#include <stdatomic.h>
#include <stdio.h>

/* A minimal mutex implemented directly on futex — identical to what
 * glibc's pthread_mutex_t does internally. */

static atomic_int lock_val = 0;   /* 0=unlocked, 1=locked */

static long futex(atomic_int *uaddr, int futex_op,
                  int val, void *timeout) {
    return syscall(SYS_futex, uaddr, futex_op, val, timeout, NULL, 0);
}

void lock() {
    int c;
    /* Fast path: if unlocked, grab it without a syscall */
    if ((c = atomic_exchange(&lock_val, 1)) == 0) return;
    /* Slow path: contended — tell kernel to sleep until woken */
    do {
        futex(&lock_val, FUTEX_WAIT, 1, NULL);  /* enter kernel, sleep */
        c = atomic_exchange(&lock_val, 1);
    } while (c != 0);
}

void unlock() {
    atomic_store(&lock_val, 0);
    futex(&lock_val, FUTEX_WAKE, 1, NULL);   /* wake one waiter */
}
/* Key insight: in the uncontended case, lock() and unlock() are
 * pure userspace atomic operations — ZERO system calls, ZERO kernel
 * transitions. The kernel is only involved when there is contention. */

Chapter 6 — Why Linux Dominates AI Infrastructure

The Six Reasons Linux Wins for AI/ML

  • GPL + Open Source: NVIDIA, AMD, and Intel can integrate their GPU drivers directly into the kernel tree. Every AI accelerator ships with a Linux driver on day one.
  • NUMA Awareness: Production GPU servers have 2–8 NUMA nodes. Linux's scheduler and memory allocator are NUMA-aware by design; Windows and macOS are not.
  • Huge Pages: LLM model weights can be 100–700 GB. Linux transparent hugepages (THP) reduce TLB pressure by 512× (4KB → 2MB pages), crucial for inference latency.
  • io_uring: Asynchronous I/O for streaming training data from NVMe at full bandwidth with near-zero CPU overhead. This directly feeds fast data loading pipelines.
  • cgroups v2: Per-process GPU memory limits, CPU quotas, and network bandwidth — essential for multi-tenant GPU clusters and Kubernetes.
  • perf + eBPF: Production profiling without recompilation. You can trace every GPU kernel launch, every memory allocation, every scheduler decision — live, in production.
The Real Numbers

On a DGX H100 (8× H100 80GB SXM5), switching from 4KB to 2MB hugepages reduces LLM inference P99 latency by 8–12% purely from TLB optimisation. On a 1,000-GPU training cluster, that's hours of compute time recovered per week.

Chapter 7 — Process Isolation for Distributed Training

How torchrun Creates Distributed Training Processes

PyTorch's torchrun uses multiprocessing.spawn to create one process per GPU. Each process gets its own rank, its own CUDA context, and communicates via NCCL (which uses shared memory + RDMA at the kernel level).

/* What torchrun does at the OS level when launching 8 GPU workers */
#define _GNU_SOURCE
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <numa.h>

#define NUM_GPUS 8

void setup_gpu_worker(int rank) {
    /* Each GPU maps to a specific NUMA node on DGX systems:
     * GPU 0-3 → NUMA 0, GPU 4-7 → NUMA 1 */
    int numa_node = rank / 4;
    pin_to_numa_node(numa_node);   /* defined in Chapter 4 */

    /* Set environment variables that NCCL and PyTorch read */
    char rank_str[4], size_str[4];
    snprintf(rank_str, sizeof(rank_str), "%d", rank);
    snprintf(size_str, sizeof(size_str), "%d", NUM_GPUS);
    setenv("RANK",       rank_str, 1);
    setenv("WORLD_SIZE", size_str, 1);
    setenv("LOCAL_RANK", rank_str, 1);

    /* Exec the actual training script */
    char *argv[] = { "python", "train.py", NULL };
    execvp("python", argv);
    perror("execvp failed");   /* only reached on error */
}

int main() {
    for (int rank = 0; rank < NUM_GPUS; rank++) {
        pid_t pid = fork();
        if (pid == 0) {        /* child */
            setup_gpu_worker(rank);
            exit(1);           /* should not reach here */
        } else if (pid < 0) {
            perror("fork"); return 1;
        }
        printf("Launched rank %d as PID %d on NUMA node %d\n",
               rank, pid, rank / 4);
    }
    /* Parent waits for all workers */
    for (int i = 0; i < NUM_GPUS; i++) {
        int status; wait(&status);
    }
    return 0;
}
