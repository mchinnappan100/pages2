Chapter 1 — Bare-Metal Cluster Setup

BIOS/UEFI Settings for GPU Servers

Before the OS even boots, BIOS/UEFI settings dramatically affect AI training performance. These settings are often undocumented and frequently wrong on freshly provisioned cloud bare-metal instances.

SettingValueWhy
Hyper-ThreadingEnabledMore OS threads → better data loading parallelism
CPU Power ManagementDisabled / Max PerformancePrevent clock throttling during sustained training
NUMA InterleavingDisabledLet Linux control NUMA allocation (we do it manually)
PCIe ASPMDisabledPCIe power saving adds latency — disable for GPUs
Above 4G DecodingEnabledRequired for GPU VRAM > 4GB BAR mapping
Resizable BAR (ReBAR)EnabledFull 80 GB VRAM visible to CPU for faster transfers
SR-IOVEnabled (if multi-tenant)Virtual function GPU partitioning

  splash \
  \

  \

  hugepagesz=1G \

  default_hugepagesz=2M \

  ### IOMMU for GPU pass-through (containers) ###
  intel_iommu=on \
  iommu=pt \               # pass-through mode: no overhead for DMA
  \
  ### GPU ###
  pci=realloc=off \        # don't move GPU BAR (can break NVLink)
  \
  ### Security (balance with performance) ###
  mitigations=auto \       # keep Spectre/Meltdown mitigations

Chapter 2 — Complete Kernel Tuning Checklist

#!/bin/bash
# COMPLETE_KERNEL_TUNING.sh
# Run on every GPU training server before a long training run.
# Estimated impact: 10-20% end-to-end training throughput improvement.
# Tested on: DGX H100, DGX A100, bare-metal 8×H100 SXM5

set -euo pipefail
echo "=== Complete Kernel Tuning for LLM Training ==="
echo "Host: $(hostname), Kernel: $(uname -r)"

## ─── 1. HUGEPAGES ──────────────────────────────────────────────── ##
echo always > /sys/kernel/mm/transparent_hugepage/enabled
echo defer+madvise > /sys/kernel/mm/transparent_hugepage/defrag
# Pre-allocate 512 GB as 1GB hugepages (for model weights + KV cache)
echo 512 > /sys/kernel/mm/hugepages/hugepages-1048576kB/nr_hugepages
echo "✓ Hugepages: $(cat /proc/meminfo | grep HugePages_Total)"

## ─── 2. VIRTUAL MEMORY ─────────────────────────────────────────── ##
sysctl -w vm.swappiness=0
sysctl -w vm.overcommit_memory=1
sysctl -w vm.dirty_ratio=80
sysctl -w vm.dirty_background_ratio=1
sysctl -w vm.max_map_count=262144    # PyTorch needs this for large models
sysctl -w kernel.perf_event_paranoid=-1  # allow perf profiling
echo "✓ VM tuning applied"

## ─── 3. NUMA ────────────────────────────────────────────────────── ##
sysctl -w kernel.numa_balancing=0
echo "✓ NUMA auto-balancing disabled"

## ─── 4. CPU PERFORMANCE ─────────────────────────────────────────── ##
for cpu in /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor; do
    echo performance > "$cpu" 2>/dev/null || true
done
echo "✓ CPU governor: performance"

## ─── 5. NETWORK (for multi-node NCCL) ──────────────────────────── ##
sysctl -w net.core.rmem_max=134217728
sysctl -w net.core.wmem_max=134217728
sysctl -w net.ipv4.tcp_rmem="4096 87380 134217728"
sysctl -w net.ipv4.tcp_wmem="4096 65536 134217728"
sysctl -w net.ipv4.tcp_mtu_probing=1
sysctl -w net.ipv4.tcp_timestamps=0   # reduce overhead
echo "✓ Network buffers expanded"

## ─── 6. NVIDIA GPU ──────────────────────────────────────────────── ##
nvidia-smi -pm 1                        # persistence mode
nvidia-smi --auto-boost-default=0       # disable boost variability
# Lock clocks on all GPUs (H100 SXM5 max: 2619 MHz memory, 1980 MHz SM)
for i in $(seq 0 $(($(nvidia-smi --list-gpus | wc -l) - 1))); do
    nvidia-smi -i "$i" -ac 2619,1980 2>/dev/null || true
done
echo "✓ GPU clocks locked: $(nvidia-smi --query-gpu=clocks.gr --format=csv,noheader | head -1) MHz"

## ─── 7. NVMe I/O SCHEDULER ─────────────────────────────────────── ##
for disk in /sys/block/nvme*; do
    echo none > "$disk/queue/scheduler" 2>/dev/null || true
    echo 1024 > "$disk/queue/nr_requests" 2>/dev/null || true
done
echo "✓ NVMe I/O scheduler: none (native queue)"

## ─── 8. OPEN FILE LIMITS ───────────────────────────────────────── ##
ulimit -n 1048576
echo "✓ Open file limit: $(ulimit -n)"

## ─── 9. VERIFY ────────────────────────────────────────────────────##
echo ""
echo "=== Verification ==="
echo "THP:         $(cat /sys/kernel/mm/transparent_hugepage/enabled)"
echo "Swappiness:  $(cat /proc/sys/vm/swappiness)"
echo "NUMA bal:    $(cat /proc/sys/kernel/numa_balancing)"
echo "GPU persist: $(nvidia-smi --query-gpu=persistence_mode --format=csv,noheader | head -1)"
echo "Free HugeP:  $(cat /proc/meminfo | grep HugePages_Free)"
echo ""
echo "✅ Tuning complete. Ready to train."

Chapter 3 — LLM Training with PyTorch FSDP

Complete Production Training Script

"""
train_llm_production.py — Complete production-grade LLM training script.
Combines: FSDP, mixed precision, gradient clipping, checkpointing,
          NCCL optimisation, and live monitoring.

Run: torchrun --nproc_per_node=8 --nnodes=4 \
              --node_rank=$RANK --master_addr=$MASTER_ADDR \
              train_llm_production.py
"""
import os, sys, time, signal, math
import torch
import torch.nn as nn
import torch.distributed as dist
from torch.distributed.fsdp import FullyShardedDataParallel as FSDP
from torch.distributed.fsdp import ShardingStrategy, MixedPrecision
from torch.distributed.fsdp.wrap import transformer_auto_wrap_policy
from pathlib import Path
import functools
import logging

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(message)s',
    handlers=[logging.StreamHandler(sys.stdout)]
)
logger = logging.getLogger(__name__)

class TrainingConfig:
    # Model
    d_model:    int   = 4096
    n_layers:   int   = 32
    n_heads:    int   = 32
    vocab_size: int   = 32000
    max_seq:    int   = 4096
    # Training
    batch_per_gpu: int   = 4
    grad_accum:    int   = 8      # effective batch = 4 × 8 × 32 GPUs = 1024
    max_steps:     int   = 100000
    lr:            float = 3e-4
    warmup_steps:  int   = 2000
    weight_decay:  float = 0.1
    grad_clip:     float = 1.0
    # Infrastructure
    checkpoint_dir: str = '/checkpoints/llm-7b'
    checkpoint_every: int = 500

def setup_distributed() -> tuple:
    dist.init_process_group(backend='nccl')
    rank       = dist.get_rank()
    world_size = dist.get_world_size()
    local_rank = int(os.environ['LOCAL_RANK'])
    torch.cuda.set_device(local_rank)
    return rank, world_size, local_rank

def cosine_lr_schedule(step: int, cfg: TrainingConfig) -> float:
    if step < cfg.warmup_steps:
        return step / cfg.warmup_steps
    progress = (step - cfg.warmup_steps) / (cfg.max_steps - cfg.warmup_steps)
    return 0.1 + 0.9 * (1 + math.cos(math.pi * progress)) / 2

def compute_mfu(loss: float, flops_per_token: int,
                tokens_per_sec: float, gpu_peak_flops: float) -> float:
    """Model FLOP Utilisation — the primary training efficiency metric.
    MFU = actual_flops / theoretical_peak_flops
    Target: > 40% for well-tuned training, > 50% is excellent."""
    actual_flops = flops_per_token * tokens_per_sec
    return actual_flops / gpu_peak_flops

def train(cfg: TrainingConfig):
    rank, world_size, local_rank = setup_distributed()
    device = torch.device(f'cuda:{local_rank}')

    if rank == 0:
        logger.info(f"Starting training: world_size={world_size}, "
                     f"effective_batch={cfg.batch_per_gpu * cfg.grad_accum * world_size}")
        Path(cfg.checkpoint_dir).mkdir(parents=True, exist_ok=True)

    # ── Model ────────────────────────────────────────────────────────
    from transformers import LlamaForCausalLM, LlamaConfig
    config = LlamaConfig(
        hidden_size=cfg.d_model, num_hidden_layers=cfg.n_layers,
        num_attention_heads=cfg.n_heads, vocab_size=cfg.vocab_size,
        max_position_embeddings=cfg.max_seq,
    )
    with torch.device('cpu'):
        model = LlamaForCausalLM(config)

    # ── FSDP wrapping ────────────────────────────────────────────────
    from transformers.models.llama.modeling_llama import LlamaDecoderLayer
    wrap_policy = functools.partial(
        transformer_auto_wrap_policy,
        transformer_layer_cls={LlamaDecoderLayer}
    )
    mp_policy = MixedPrecision(
        param_dtype=torch.bfloat16,
        reduce_dtype=torch.float32,
        buffer_dtype=torch.bfloat16,
    )
    model = FSDP(
        model,
        sharding_strategy=ShardingStrategy.FULL_SHARD,
        auto_wrap_policy=wrap_policy,
        mixed_precision=mp_policy,
        device_id=torch.cuda.current_device(),
        limit_all_gathers=True,
    )

    # ── Optimiser ────────────────────────────────────────────────────
    optimiser = torch.optim.AdamW(
        model.parameters(), lr=cfg.lr,
        weight_decay=cfg.weight_decay, betas=(0.9, 0.95), eps=1e-8
    )
    scheduler = torch.optim.lr_scheduler.LambdaLR(
        optimiser, lambda step: cosine_lr_schedule(step, cfg)
    )

    # ── Resume from checkpoint ───────────────────────────────────────
    start_step = 0
    ckpts = sorted(Path(cfg.checkpoint_dir).glob('step_*.pt'))
    if ckpts:
        ckpt = torch.load(ckpts[-1], map_location='cpu')
        model.load_state_dict(ckpt['model'])
        optimiser.load_state_dict(ckpt['optimiser'])
        start_step = ckpt['step']
        if rank == 0: logger.info(f"Resumed from step {start_step}")

    # ── SIGTERM handler (spot instance preemption) ───────────────────
    def _save_and_exit(signum, frame):
        if rank == 0:
            torch.save({'step': step, 'model': model.state_dict(),
                         'optimiser': optimiser.state_dict()},
                        f'{cfg.checkpoint_dir}/emergency.pt')
        dist.destroy_process_group()
        sys.exit(0)
    signal.signal(signal.SIGTERM, _save_and_exit)

    # ── Training loop ────────────────────────────────────────────────
    model.train()
    H100_PEAK_FLOPS = 989e12  # BF16 non-Tensor-Core

    for step in range(start_step, cfg.max_steps):
        t0 = time.perf_counter()
        total_loss = 0.0
        tokens_this_step = 0

        for micro_step in range(cfg.grad_accum):
            # Synthetic batch — replace with real DataLoader in production
            input_ids = torch.randint(0, cfg.vocab_size,
                              (cfg.batch_per_gpu, cfg.max_seq), device=device)
            labels    = input_ids.clone()

            with torch.cuda.amp.autocast(dtype=torch.bfloat16):
                outputs = model(input_ids=input_ids, labels=labels)
                loss    = outputs.loss / cfg.grad_accum

            loss.backward()
            total_loss    += loss.item()
            tokens_this_step += cfg.batch_per_gpu * cfg.max_seq

        # Gradient clipping + optimiser step
        nn.utils.clip_grad_norm_(model.parameters(), cfg.grad_clip)
        optimiser.step()
        scheduler.step()
        optimiser.zero_grad(set_to_none=True)

        if rank == 0 and step % 10 == 0:
            dt          = time.perf_counter() - t0
            tokens_sec  = tokens_this_step * world_size / dt
            # ~6 × params FLOPs per token (forward + backward)
            params      = sum(p.numel() for p in model.parameters())
            mfu         = compute_mfu(total_loss, 6 * params, tokens_sec,
                                       H100_PEAK_FLOPS * world_size)
            logger.info(
                f"step={step:6d} loss={total_loss:.4f} "
                f"lr={scheduler.get_last_lr()[0]:.2e} "
                f"tokens/s={tokens_sec/1000:.1f}K "
                f"MFU={mfu*100:.1f}%"
            )

        if rank == 0 and step % cfg.checkpoint_every == 0:
            torch.save({'step': step, 'model': model.state_dict(),
                         'optimiser': optimiser.state_dict()},
                        f'{cfg.checkpoint_dir}/step_{step:08d}.pt')

    dist.destroy_process_group()

if __name__ == '__main__':
    train(TrainingConfig())

Chapter 4 — Debugging Training Failures

"""
debug_training.py — Systematic debug checklist when training crashes.
"""
import torch, os, subprocess

def full_diagnostic():
    print("=== Training Failure Diagnostic ===")

    # 1. Check GPU health
    print("\n1. GPU Health:")
    r = subprocess.run(['nvidia-smi', '-q', '-d', 'ECC'], capture_output=True, text=True)
    ecc_lines = [l for l in r.stdout.split('\n') if 'Volatile' in l or 'Aggregate' in l]
    print('\n'.join(ecc_lines[:10]) or "  No ECC errors")

    # 2. Check for NaN/Inf in recent checkpoint
    print("\n2. Checkpoint Integrity:")
    from pathlib import Path
    ckpts = sorted(Path('/checkpoints').glob('**/*.pt'))
    if ckpts:
        ckpt = torch.load(ckpts[-1], map_location='cpu')
        has_nan = any(torch.isnan(v).any() for v in ckpt.get('model', {}).values()
                     if isinstance(v, torch.Tensor))
        print(f"  Latest: {ckpts[-1].name}, NaN: {has_nan}")

    # 3. Check kernel OOM kills
    print("\n3. OOM Kill History:")
    r = subprocess.run(['dmesg', '--level=err'], capture_output=True, text=True)
    oom = [l for l in r.stdout.split('\n') if 'oom' in l.lower() or 'killed' in l.lower()]
    print('\n'.join(oom[-5:]) or "  No OOM kills found")

    # 4. Check NCCL errors
    print("\n4. NCCL Status:")
    print(f"  NCCL_DEBUG={os.environ.get('NCCL_DEBUG', 'not set')}")
    print(f"  NCCL_IB_DISABLE={os.environ.get('NCCL_IB_DISABLE', 'not set')}")
    print(f"  NCCL_P2P_LEVEL={os.environ.get('NCCL_P2P_LEVEL', 'not set')}")

    # 5. GPU memory state
    print("\n5. CUDA Memory:")
    for i in range(torch.cuda.device_count()):
        used = torch.cuda.memory_allocated(i) / 1e9
        total = torch.cuda.get_device_properties(i).total_memory / 1e9
        print(f"  GPU {i}: {used:.1f}/{total:.0f} GB allocated")

if __name__ == '__main__':
    full_diagnostic()

Production Observability

"""
vllm_monitor.py — Real-time vLLM serving metrics.
Connects to the vLLM metrics endpoint and displays key serving KPIs.
"""
import requests, time

def monitor_vllm(endpoint: str = "http://localhost:8000"):
    while True:
        try:
            metrics = requests.get(f"{endpoint}/metrics", timeout=5)
            text = metrics.text

            def extract(name):
                for line in text.split('\n'):
                    if line.startswith(name + ' '):
                        try: return float(line.split()[1])
                        except: return 0
                return 0

            pending     = extract('vllm:num_requests_waiting')
            running     = extract('vllm:num_requests_running')
            gpu_cache   = extract('vllm:gpu_cache_usage_perc')
            tput        = extract('vllm:avg_generation_throughput')
            ttft_ms     = extract('vllm:time_to_first_token_seconds') * 1000
            tpot_ms     = extract('vllm:time_per_output_token_seconds') * 1000

            print(f"\r[vLLM] running={running:.0f} "
                  f"pending={pending:.0f} "
                  f"gpu_kv={gpu_cache*100:.1f}% "
                  f"tput={tput:.0f}tok/s "
                  f"TTFT={ttft_ms:.0f}ms "
                  f"TPOT={tpot_ms:.1f}ms", end='')
        except Exception as e:
            print(f"\rConnection error: {e}", end='')
        time.sleep(2)

if __name__ == '__main__':
    monitor_vllm()

Chapter 6 — eBPF Monitoring in Production

"""
ebpf_ai_monitor.py — Production eBPF monitoring for AI workloads.
Uses BCC (BPF Compiler Collection) for Python-friendly eBPF programs.
Monitors: GPU memory allocations, NCCL AllReduce latency,
          DataLoader read throughput, OOM events.
Install: pip install bcc (requires kernel headers)
"""
from bcc import BPF
import time, ctypes

BPF_PROGRAM = """
#include 
#include 

// Track NCCL AllReduce start times
BPF_HASH(allreduce_start, u32, u64);  // tid -> start_ns

// Record completed AllReduce latencies
BPF_HISTOGRAM(allreduce_latency_us);

// Count OOM events
BPF_ARRAY(oom_count, u32, 1);

// Track large memory allocations (> 1 GB)
BPF_PERF_OUTPUT(large_allocs);
struct alloc_event {
    u32 pid;
    u64 size;
    char comm[16];
};

// Probe NCCL AllReduce entry
int probe_allreduce_entry(struct pt_regs *ctx) {
    u32 tid = bpf_get_current_pid_tgid();
    u64 ts  = bpf_ktime_get_ns();
    allreduce_start.update(&tid, &ts);
    return 0;
}

// Probe NCCL AllReduce return — compute latency
int probe_allreduce_return(struct pt_regs *ctx) {
    u32 tid = bpf_get_current_pid_tgid();
    u64 *start = allreduce_start.lookup(&tid);
    if (!start) return 0;
    u64 latency_us = (bpf_ktime_get_ns() - *start) / 1000;
    allreduce_latency_us.increment(bpf_log2l(latency_us));
    allreduce_start.delete(&tid);
    return 0;
}

// Trace OOM kills
int trace_oom_kill(struct pt_regs *ctx) {
    u32 key = 0;
    u32 *count = oom_count.lookup(&key);
    if (count) (*count)++;
    return 0;
}
"""

b = BPF(text=BPF_PROGRAM)

# Attach to NCCL AllReduce (uprobe on libncclknl.so)
try:
    b.attach_uprobe(name="nccl", sym="ncclAllReduce",
                    fn_name="probe_allreduce_entry")
    b.attach_uretprobe(name="nccl", sym="ncclAllReduce",
                      fn_name="probe_allreduce_return")
    print("NCCL AllReduce probes attached")
except:
    print("NCCL probes not available (NCCL not running)")

# Attach to kernel OOM kill function
b.attach_kprobe(event="oom_kill_process", fn_name="trace_oom_kill")

print("Monitoring AI workload... Press Ctrl+C to print histogram")

try:
    while True:
        time.sleep(10)
        oom_val = b["oom_count"][ctypes.c_int(0)].value
        print(f"OOM kills: {oom_val}")
except KeyboardInterrupt:
    print("\nNVCCL AllReduce Latency Distribution (µs):")
    b["allreduce_latency_us"].print_log2_hist("latency (µs)")

#!/bin/bash
# 3am_debug.sh — When training crashes and you need answers fast

echo "=== 3 AM LLM Training Incident Response ==="

# Q1: Is the GPU dead?
echo "[1] GPU Health:"
nvidia-smi --query-gpu=index,ecc.errors.corrected.volatile.total,\
ecc.errors.uncorrected.volatile.total,temperature.gpu --format=csv

# Q2: Was there an OOM kill?
echo "[2] OOM Kills (last 10 min):"
dmesg --since "-10min" | grep -i "oom\|killed" | tail -10
