Why This Book Exists

Training GPT-3 required 1,024 A100s running for 34 days. Training Llama-3 405B required several thousand H100s. At that scale, every Linux configuration decision — NUMA affinity, network interrupt coalescing, cgroup limits, huge page settings — has measurable impact on training throughput.

Chapter 1 — Process Groups & NCCL

PyTorch Process Groups

A process group is a set of processes that can communicate with each other via collective operations. Each process in the group is identified by its rank (0 to world_size-1). Process groups are the abstraction layer over NCCL's actual inter-GPU communication.

"""
process_groups.py — Understanding PyTorch distributed process groups.
Run: torchrun --nproc_per_node=8 process_groups.py
"""
import os
import torch
import torch.distributed as dist

def init_process_group():
    # torchrun sets these environment variables automatically
    rank       = int(os.environ['RANK'])
    world_size = int(os.environ['WORLD_SIZE'])
    local_rank = int(os.environ['LOCAL_RANK'])

    # Set this GPU as the default device for this process
    torch.cuda.set_device(local_rank)

    # Initialize the default process group using NCCL backend
    # NCCL = NVIDIA Collective Communications Library
    # gloo = CPU-based fallback (for debugging without GPU)
    dist.init_process_group(
        backend='nccl',
        init_method='env://',   # reads MASTER_ADDR, MASTER_PORT from env
    )
    print(f"[Rank {rank}/{world_size}] GPU={local_rank} initialized")
    return rank, world_size, local_rank

def create_subgroups(rank, world_size):
    """
    Create specialised sub-groups for tensor parallelism.
    On an 8-GPU DGX node:
    - Data parallel group: ranks within the same TP group
    - Tensor parallel group: ranks across DP groups
    """
    TP_SIZE = 4   # tensor parallel degree
    DP_SIZE = world_size // TP_SIZE   # data parallel degree

    # Create tensor parallel groups: (0,1,2,3), (4,5,6,7)
    tp_groups = []
    for dp_rank in range(DP_SIZE):
        ranks = list(range(dp_rank * TP_SIZE, (dp_rank + 1) * TP_SIZE))
        grp   = dist.new_group(ranks=ranks, backend='nccl')
        tp_groups.append((ranks, grp))
        if rank in ranks:
            my_tp_group = grp
            print(f"[Rank {rank}] TP group: {ranks}")

    # Create data parallel groups: (0,4), (1,5), (2,6), (3,7)
    for tp_rank in range(TP_SIZE):
        ranks = list(range(tp_rank, world_size, TP_SIZE))
        grp   = dist.new_group(ranks=ranks, backend='nccl')
        if rank in ranks:
            my_dp_group = grp
            print(f"[Rank {rank}] DP group: {ranks}")

    return my_tp_group, my_dp_group

def demo_collectives(rank, world_size):
    """Demonstrate NCCL collective operations used in training."""
    device = torch.cuda.current_device()
    tensor = torch.tensor([rank * 1.0], device=device)

    # AllReduce: sum across all GPUs — used for gradient synchronisation
    dist.all_reduce(tensor, op=dist.ReduceOp.SUM)
    expected = sum(range(world_size))  # 0+1+...+(n-1)
    print(f"[{rank}] AllReduce result: {tensor.item():.0f} (expected {expected})")

    # AllGather: each rank shares its tensor with all others
    my_data  = torch.tensor([rank * 2.0], device=device)
    gathered = [torch.zeros(1, device=device) for _ in range(world_size)]
    dist.all_gather(gathered, my_data)
    if rank == 0:
        print(f"AllGather: {[t.item() for t in gathered]}")

    # Broadcast: rank 0 sends to all — used for model weight initialisation
    model_hash = torch.tensor([42.0 if rank == 0 else 0.0], device=device)
    dist.broadcast(model_hash, src=0)
    print(f"[{rank}] Broadcast: {model_hash.item()}")  # all print 42.0

if __name__ == '__main__':
    rank, world_size, local_rank = init_process_group()
    demo_collectives(rank, world_size)
    dist.destroy_process_group()

NCCL Internals — What Happens Under the Hood

When your PyTorch code calls dist.all_reduce(), NCCL executes the following sequence at the Linux kernel level:

  1. Checks the communication topology (NVLink, PCIe, or IB)
  2. For NVLink-connected GPUs on the same node: uses direct GPU-to-GPU P2P transfers via mapped BAR memory
  3. For multi-node: uses RDMA write operations over InfiniBand (bypassing the CPU entirely)
  4. Uses the ring-allreduce or tree-allreduce algorithm to minimise total data transferred
# Critical NCCL environment variables for production training

# Use NVLink for intra-node GPU communication (default on DGX)
export NCCL_P2P_LEVEL=NVL

# Use InfiniBand for inter-node communication
export NCCL_IB_DISABLE=0
export NCCL_IB_HCA=mlx5_0:1  # specify which IB HCA to use

# Ring-allreduce chunking — larger = better BW utilisation, more latency
export NCCL_BUFFSIZE=4194304  # 4 MB chunks

# Disable socket fallback (use only IB/NVLink)
export NCCL_SOCKET_IFNAME=^lo,docker0

# Debug: show which transport NCCL chose
export NCCL_DEBUG=INFO
export NCCL_DEBUG_SUBSYS=INIT,NET

Chapter 2 — cgroups v2 for GPU Isolation

Memory Limits with cgroups v2

cgroups v2 (control groups) is the Linux kernel mechanism for limiting and isolating resources per process group. On a multi-tenant GPU cluster, cgroups prevent one training job from consuming all system memory and killing other jobs via OOM.

/* Inspecting cgroup memory limits from a C program.
 * This is what Kubernetes does when enforcing pod memory limits. */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

long get_cgroup_memory_limit(void) {
    FILE *f = fopen("/sys/fs/cgroup/memory.max", "r");
    if (!f) return -1;
    char buf[32];
    fgets(buf, sizeof(buf), f);
    fclose(f);
    if (strncmp(buf, "max", 3) == 0) return -1;  /* no limit */
    return atol(buf);
}

long get_cgroup_memory_current(void) {
    FILE *f = fopen("/sys/fs/cgroup/memory.current", "r");
    if (!f) return -1;
    char buf[32];
    fgets(buf, sizeof(buf), f);
    fclose(f);
    return atol(buf);
}

int main(void) {
    long limit   = get_cgroup_memory_limit();
    long current = get_cgroup_memory_current();

    if (limit < 0) {
        printf("Memory limit: unlimited\n");
    } else {
        printf("Memory limit:   %ld GB\n", limit / (1024*1024*1024));
        printf("Memory current: %ld GB\n", current / (1024*1024*1024));
        printf("Memory usage:   %.1f%%\n",
               100.0 * current / limit);
    }
    return 0;
}

# setup_training_cgroup.sh — Create a cgroup for an LLM training job
# Usage: ./setup_training_cgroup.sh  

TRAINING_PID=$1
MEM_LIMIT_GB=$2
CGROUP_NAME="llm_training_$$"
CGROUP_PATH="/sys/fs/cgroup/$CGROUP_NAME"

# Create the cgroup

# Move training process into the cgroup
echo "$TRAINING_PID" > "$CGROUP_PATH/cgroup.procs"

echo "  Memory limit: ${MEM_LIMIT_GB} GB"
echo "  CPU weight:   200 (2× default priority)"
echo "  Swap:         disabled"

Chapter 3 — DDP: Data Parallel Training

AllReduce — How Gradient Synchronisation Works

In data parallel training, each GPU computes gradients on its own mini-batch. Before the optimiser step, all gradients must be averaged across all GPUs. This is done with a Ring-AllReduce algorithm that communicates O(N) data per GPU, not O(N²).

OperationCommunication PatternUsed For
AllReduce (sum)Ring: each GPU sends and receives onceGradient averaging in DDP
AllGatherEach rank gets full tensorGathering weight shards in FSDP
ReduceScatterEach rank gets a slice of the reduced tensorDistributing averaged gradients in ZeRO
BroadcastOne rank sends to allWeight initialisation, checkpoint load
AllToAllEach rank sends unique data to each other rankExpert routing in Mixture-of-Experts

PyTorch DDP Training Loop

"""
ddp_training.py — Complete DDP training loop for LLM pre-training.
Run: torchrun --nproc_per_node=8 ddp_training.py
"""
import os
import torch
import torch.nn as nn
import torch.distributed as dist
from torch.nn.parallel import DistributedDataParallel as DDP
from torch.utils.data import DataLoader, DistributedSampler

def setup():
    dist.init_process_group(backend='nccl')
    local_rank = int(os.environ['LOCAL_RANK'])
    torch.cuda.set_device(local_rank)
    return local_rank

def cleanup():
    dist.destroy_process_group()

def train(model_class, dataset, config):
    local_rank = setup()
    rank       = dist.get_rank()
    world_size = dist.get_world_size()
    device     = torch.device(f'cuda:{local_rank}')

    # Build model and move to GPU
    model = model_class(config).to(device)

    # Wrap with DDP — this inserts AllReduce hooks into the backward pass
    model = DDP(
        model,
        device_ids=[local_rank],
        output_device=local_rank,
        find_unused_parameters=False,  # set True only if needed — adds overhead
        gradient_as_bucket_view=True,   # reduce memory by reusing gradient buffers
    )

    # DistributedSampler ensures each GPU sees different data each epoch
    sampler    = DistributedSampler(dataset, num_replicas=world_size, rank=rank)
    dataloader = DataLoader(
        dataset, batch_size=config.batch_per_gpu,
        sampler=sampler, num_workers=4, pin_memory=True
    )

    optimiser = torch.optim.AdamW(model.parameters(), lr=config.lr,
                                   weight_decay=0.1)
    scaler    = torch.cuda.amp.GradScaler()  # mixed precision

    for epoch in range(config.epochs):
        sampler.set_epoch(epoch)   # crucial: shuffles differently each epoch

        for step, batch in enumerate(dataloader):
            input_ids = batch['input_ids'].to(device)
            labels    = batch['labels'].to(device)

            optimiser.zero_grad()

            # Mixed precision forward pass
            with torch.cuda.amp.autocast(dtype=torch.bfloat16):
                logits = model(input_ids)
                loss   = nn.functional.cross_entropy(
                    logits.view(-1, config.vocab_size),
                    labels.view(-1)
                )

            # Backward — DDP automatically AllReduces gradients here
            scaler.scale(loss).backward()

            # Gradient clipping (critical for LLM training stability)
            scaler.unscale_(optimiser)
            nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)

            scaler.step(optimiser)
            scaler.update()

            if rank == 0 and step % 10 == 0:
                print(f"Epoch {epoch} Step {step} Loss {loss.item():.4f}")

    cleanup()

Chapter 4 — FSDP: Fully Sharded Data Parallelism

ZeRO Memory Optimisation Stages

Standard DDP keeps a full copy of model weights, gradients, and optimiser state on every GPU. For a 70B model in BF16 + Adam: ~140 GB weights + ~280 GB optimiser state = 420 GB per GPU — impossible on a single 80 GB H100.

ZeRO (Zero Redundancy Optimizer) shards these across GPUs:

ZeRO StageWhat Gets ShardedMemory Reduction
Stage 1Optimiser states only~4× reduction
Stage 2Optimiser states + gradients~8× reduction
Stage 3 (FSDP)Optimiser states + gradients + parameters~N× reduction (N = GPU count)

FSDP Training Loop

"""
fsdp_training.py — FSDP (ZeRO Stage 3) for training models that don't
fit on a single GPU. Run: torchrun --nproc_per_node=8 fsdp_training.py
"""
import os, torch, torch.nn as nn, torch.distributed as dist
from torch.distributed.fsdp import (
    FullyShardedDataParallel as FSDP,
    ShardingStrategy,
    BackwardPrefetch,
    MixedPrecision,
    CPUOffload,
)
from torch.distributed.fsdp.wrap import transformer_auto_wrap_policy
import functools

def build_fsdp_model(model_class, config, local_rank):
    """Wrap a transformer model with FSDP for sharded training."""
    device = torch.device(f'cuda:{local_rank}')

    # Build model on CPU first to avoid OOM during initialisation
    with torch.device('cpu'):
        model = model_class(config)

    # Mixed precision policy: BF16 for compute, FP32 for parameter storage
    mp_policy = MixedPrecision(
        param_dtype=torch.bfloat16,    # parameters stored in BF16
        reduce_dtype=torch.float32,    # gradients reduced in FP32 for stability
        buffer_dtype=torch.bfloat16,   # buffers (LayerNorm, etc.) in BF16
    )

    # Auto wrap policy: shard at transformer layer boundaries
    # Each TransformerBlock becomes an independent FSDP unit
    wrap_policy = functools.partial(
        transformer_auto_wrap_policy,
        transformer_layer_cls={config.transformer_layer_class}
    )

    model = FSDP(
        model,
        sharding_strategy=ShardingStrategy.FULL_SHARD,  # ZeRO Stage 3
        auto_wrap_policy=wrap_policy,
        mixed_precision=mp_policy,
        backward_prefetch=BackwardPrefetch.BACKWARD_PRE,  # overlap compute+comm
        device_id=torch.cuda.current_device(),
        cpu_offload=None,   # CPUOffload() to offload params to CPU if needed
        limit_all_gathers=True,  # prevent memory spikes from too many all-gathers
    )
    print(f"[Rank {dist.get_rank()}] FSDP model built. "
          f"Local params: {sum(p.numel() for p in model.parameters()):,}")
    return model

def save_fsdp_checkpoint(model, optimiser, step: int, rank: int):
    """Save FSDP checkpoint — must be done while model is fully materialised."""
    from torch.distributed.fsdp import FullStateDictConfig, StateDictType
    from torch.distributed.fsdp import FullyShardedDataParallel as FSDP

    # Gather full state dict on rank 0 only
    save_policy = FullStateDictConfig(offload_to_cpu=True, rank0_only=True)
    with FSDP.state_dict_type(model, StateDictType.FULL_STATE_DICT, save_policy):
        state = model.state_dict()

    if rank == 0:
        torch.save({
            'step':      step,
            'model':     state,
            'optimiser': optimiser.state_dict(),
        }, f'checkpoint_step_{step}.pt')
        print(f"Checkpoint saved at step {step}")

Chapter 5 — Network Fabric & RDMA

Multi-node LLM training requires connecting hundreds of servers with a network fast enough to keep GPUs busy during AllReduce. InfiniBand with RDMA is the standard choice.

TechnologyBandwidthLatencyCPU Overhead
100 GbE (Ethernet)100 Gb/s~5 µsHigh
200 Gb HDR InfiniBand200 Gb/s~0.5 µsNear-zero (RDMA)
400 Gb NDR InfiniBand400 Gb/s~0.5 µsNear-zero (RDMA)
NVLink (intra-node)900 GB/s~1 µsNear-zero
/* RDMA (Remote Direct Memory Access) — the kernel bypass technology
 * that lets NCCL write directly to another server's GPU memory
 * without involving either machine's CPU.
 *
 * The key system calls that enable RDMA on InfiniBand:
 */
#include <infiniband/verbs.h>

/* Step 1: Register memory region — pin it and get a DMA address */
struct ibv_mr *register_rdma_buffer(struct ibv_pd *pd,
                                     void *buf, size_t size)
{
    /* IBV_ACCESS_REMOTE_WRITE: allow remote peers to write to this buffer.
     * This is how NCCL receives gradient updates from other nodes. */
    return ibv_reg_mr(pd, buf, size,
                       IBV_ACCESS_LOCAL_WRITE  |
                       IBV_ACCESS_REMOTE_WRITE |
                       IBV_ACCESS_REMOTE_READ);
}

/* Step 2: Post an RDMA Write — sends data to remote buffer without
 * touching the remote CPU at all.
 * NCCL uses this for AllReduce: write gradients directly to
 * the next node's GPU buffer in the ring. */
int rdma_write_gradient_shard(struct ibv_qp *qp,
                               void *local_buf, size_t size,
                               uint32_t local_lkey,
                               uint64_t remote_addr,
                               uint32_t remote_rkey)
{
    struct ibv_sge sg = {
        .addr   = (uintptr_t)local_buf,
        .length = size,
        .lkey   = local_lkey,
    };
    struct ibv_send_wr wr = {
        .wr_id      = 1,
        .sg_list    = &sg,
        .num_sge    = 1,
        .opcode     = IBV_WR_RDMA_WRITE,   /* write to remote memory */
        .send_flags = IBV_SEND_SIGNALED,
        .wr.rdma = {
            .remote_addr = remote_addr,    /* destination GPU buffer address */
            .rkey        = remote_rkey,    /* remote memory key */
        },
    };
    struct ibv_send_wr *bad_wr;
    return ibv_post_send(qp, &wr, &bad_wr);
    /* CPU is now free — IB HCA DMA engine handles the transfer */
}

Chapter 6 — Linux Kernel Tuning Checklist

#!/bin/bash
# kernel_tuning.sh — Complete Linux kernel tuning for GPU training servers
# Run as root: sudo bash kernel_tuning.sh

## 1. HUGE PAGES — reduces TLB pressure for large model tensors ##
echo always > /sys/kernel/mm/transparent_hugepage/enabled
echo defer+madvise > /sys/kernel/mm/transparent_hugepage/defrag
# For explicit huge pages (most stable):
# echo 4096 > /sys/kernel/mm/hugepages/hugepages-2048kB/nr_hugepages

## 2. VM SETTINGS — prevent OOM-killer from killing training jobs ##
sysctl -w vm.swappiness=10        # rarely swap (0=never, 60=default)
sysctl -w vm.dirty_ratio=80       # allow large dirty page cache
sysctl -w vm.dirty_background_ratio=5
sysctl -w vm.overcommit_memory=1  # allow memory overcommit (needed for fork)
sysctl -w vm.overcommit_ratio=95

## 3. NETWORK — optimize for RDMA and NCCL ##
sysctl -w net.core.rmem_max=134217728        # 128 MB socket receive buffer
sysctl -w net.core.wmem_max=134217728
sysctl -w net.ipv4.tcp_rmem="4096 87380 134217728"
sysctl -w net.ipv4.tcp_wmem="4096 65536 134217728"
sysctl -w net.core.netdev_max_backlog=250000

## 4. NUMA — prevent automatic NUMA balancing (we do it manually) ##
echo 0 > /proc/sys/kernel/numa_balancing

## 5. CPU FREQUENCY GOVERNOR — maximum performance ##
for cpu in /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor; do
    echo performance > "$cpu" 2>/dev/null
done

## 6. IRQ AFFINITY — pin network IRQs to CPUs near IB HCA ##
# Find IB HCA IRQs and pin to NUMA node 0 CPUs
for irq in $(grep mlx5 /proc/interrupts | awk -F: '{print $1}'); do
    echo ff > /proc/irq/"$irq"/smp_affinity  # CPUs 0-7
done

## 7. NVIDIA GPU SETTINGS ##
nvidia-smi -pm 1                          # persistence mode: keep GPU initialised
nvidia-smi --auto-boost-default=0         # disable boost clock variability
nvidia-smi -ac 2619,1980                  # H100: lock memory,graphics clocks

## 8. OPEN FILE LIMITS — torchrun opens many file descriptors ##
ulimit -n 1048576
echo "* soft nofile 1048576" >> /etc/security/limits.conf
echo "* hard nofile 1048576" >> /etc/security/limits.conf

echo "Kernel tuning complete. Estimated improvement: 10-15% training throughput."

Chapter 7 — Fault Tolerance for Long Training Runs

"""
fault_tolerance.py — Checkpoint and recovery for multi-day LLM training runs.
A 70B model training run takes weeks — nodes will fail. You must be ready.
"""
import os, signal, torch, torch.distributed as dist
from pathlib import Path
from datetime import datetime

class RobustTrainer:
    CHECKPOINT_INTERVAL = 500   # save every 500 steps

    def __init__(self, model, optimiser, checkpoint_dir: str):
        self.model         = model
        self.optimiser     = optimiser
        self.checkpoint_dir = Path(checkpoint_dir)
        self.checkpoint_dir.mkdir(exist_ok=True)
        self.step          = 0
        self.rank          = dist.get_rank() if dist.is_initialized() else 0

        # Handle SIGTERM gracefully — cloud providers send this before killing nodes
        signal.signal(signal.SIGTERM, self._handle_sigterm)

    def _handle_sigterm(self, signum, frame):
        print(f"[Rank {self.rank}] SIGTERM received — saving emergency checkpoint")
        self.save_checkpoint(emergency=True)
        dist.destroy_process_group()
        exit(0)

    def save_checkpoint(self, emergency: bool = False):
        if self.rank != 0:
            dist.barrier(); return   # only rank 0 saves
        tag = "emergency" if emergency else f"step_{self.step:08d}"
        path = self.checkpoint_dir / f"ckpt_{tag}.pt"
        torch.save({
            'step':      self.step,
            'model':     self.model.state_dict(),
            'optimiser': self.optimiser.state_dict(),
            'timestamp': datetime.now().isoformat(),
        }, path)
        print(f"Checkpoint saved: {path} ({path.stat().st_size / 1e9:.1f} GB)")

        # Keep only last 3 checkpoints to save disk space
        ckpts = sorted(self.checkpoint_dir.glob("ckpt_step_*.pt"))
        for old in ckpts[:-3]:
            old.unlink()
        dist.barrier()

    def load_latest_checkpoint(self) -> bool:
        ckpts = sorted(self.checkpoint_dir.glob("ckpt_step_*.pt"))
        if not ckpts:
            print("No checkpoint found — starting from scratch")
            return False
        ckpt = torch.load(ckpts[-1], map_location='cpu')
        self.model.load_state_dict(ckpt['model'])
        self.optimiser.load_state_dict(ckpt['optimiser'])
        self.step = ckpt['step']
        print(f"Resumed from step {self.step} ({ckpts[-1].name})")
        return True

    def training_step(self, batch):
        # ... actual training logic ...
        self.step += 1
        if self.step % self.CHECKPOINT_INTERVAL == 0:
            self.save_checkpoint()

Production Checklist for Long Training Runs

Before starting a multi-week run: ① Test checkpoint save/load in a short dry run. ② Set up automatic alerting on node failures. ③ Enable SIGTERM handler for graceful spot-instance preemption. ④ Store checkpoints on a shared NFS/Lustre filesystem accessible from all nodes. ⑤ Log MFU (Model FLOP Utilisation) every 100 steps — a drop signals a bottleneck.
