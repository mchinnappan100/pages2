Why Device Drivers Matter for AI Engineers

The NVIDIA GPU driver (nvidia.ko) is the most performance-critical piece of software on your AI training server. Understanding how Linux device drivers work lets you diagnose driver hangs, understand CUDA's memory model at the kernel level, configure DMA correctly for fast data paths, and write custom accelerator drivers for custom silicon (Google TPUs, Cerebras, Groq all started with a custom Linux driver).

What You Will Build

In Chapter 4 we build a complete, working temperature monitor character device driver from scratch — with I2C sensor communication, hwmon subsystem integration, interrupt handling, and a user-space Python client to read temperatures. This same pattern is the foundation of every hardware driver.

Kernel Module Anatomy

Every kernel module is a .ko ELF object that is loaded into kernel space at runtime. The minimal module requires only two functions: an init function called on insmod and an exit function called on rmmod.

/* hello_driver.c — The simplest possible kernel module.
 * Build: make -C /lib/modules/$(uname -r)/build M=$(pwd) modules
 * Load:  sudo insmod hello_driver.ko
 * Unload:sudo rmmod hello_driver
 * Logs:  sudo dmesg | tail -5 */

#include <linux/init.h>
#include <linux/module.h>
#include <linux/kernel.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("AI Driver Engineer");
MODULE_DESCRIPTION("Hello World kernel module");
MODULE_VERSION("1.0");

static int __init hello_init(void)
{
    printk(KERN_INFO "[hello] Driver loaded. "
           "This runs in Ring 0, kernel space.\n");
    return 0;   /* 0 = success; negative errno = failure */
}

static void __exit hello_exit(void)
{
    printk(KERN_INFO "[hello] Driver unloaded. Cleaning up.\n");
}

module_init(hello_init);
module_exit(hello_exit);

Chapter 2 — Character Device Drivers

The cdev Structure and file_operations

A character device driver exposes a file-like interface via /dev/yourdevice. User space opens it, reads/writes bytes, and uses ioctl() to send commands. The kernel dispatches these operations through a file_operations function pointer table.

/* The file_operations struct — the jump table between VFS and your driver.
 * You implement only the operations you support; leave others NULL. */
static const struct file_operations mydev_fops = {
    .owner          = THIS_MODULE,
    .open           = mydev_open,
    .release        = mydev_release,
    .read           = mydev_read,
    .write          = mydev_write,
    .unlocked_ioctl = mydev_ioctl,
    .mmap           = mydev_mmap,   /* GPU drivers implement this for VRAM mapping */
    .poll           = mydev_poll,
};

The ioctl Interface — How CUDA Talks to the GPU Driver

The ioctl() system call is the primary control channel between the CUDA runtime and the nvidia.ko driver. Every cudaMalloc(), cuLaunchKernel(), and cudaMemcpy() translates to one or more ioctl() calls on /dev/nvidia0.

/* Defining ioctl commands for a custom driver.
 * Use the _IOW/_IOR/_IOWR macros to encode type, number, and size. */

#define MYDEV_IOC_MAGIC  'M'

/* Get status — kernel → user */
#define MYDEV_GET_STATUS  _IOR(MYDEV_IOC_MAGIC, 0, struct mydev_status)

/* Set configuration — user → kernel */
#define MYDEV_SET_CONFIG  _IOW(MYDEV_IOC_MAGIC, 1, struct mydev_config)

/* Submit work — user → kernel (returns completion token) */
#define MYDEV_SUBMIT_JOB  _IOWR(MYDEV_IOC_MAGIC, 2, struct mydev_job)

struct mydev_status {
    __u32 temperature_mc;   /* milli-Celsius */
    __u32 load_percent;
    __u64 bytes_processed;
};

/* In the driver's unlocked_ioctl handler: */
static long mydev_ioctl(struct file *file, unsigned int cmd,
                         unsigned long arg)
{
    struct mydev_priv *priv = file->private_data;

    switch (cmd) {
    case MYDEV_GET_STATUS: {
        struct mydev_status status;
        mydev_read_hardware_status(priv, &status);
        /* copy_to_user: safely copies kernel data to user address space */
        if (copy_to_user((void __user *)arg, &status, sizeof(status)))
            return -EFAULT;
        return 0;
    }
    case MYDEV_SET_CONFIG: {
        struct mydev_config cfg;
        if (copy_from_user(&cfg, (const void __user *)arg, sizeof(cfg)))
            return -EFAULT;
        return mydev_apply_config(priv, &cfg);
    }
    default:
        return -ENOTTY;   /* "not a typewriter" — unknown ioctl */
    }
}

Chapter 3 — Hardware Basics: I2C & SPI

Most on-board sensors (temperature, voltage, power) use I2C (Inter-Integrated Circuit) — a two-wire serial bus. In a GPU server, the baseboard management controller (BMC) uses I2C to read temperature sensors across the board, and your driver can too.

ProtocolWiresSpeedCommon Use
I2CSCL + SDA (2)100 kHz – 400 kHzTemperature, voltage, EEPROM sensors
SPISCK + MOSI + MISO + CS (4)1–100 MHzFlash memory, ADCs, high-speed sensors
SMBusSCL + SDA (2)100 kHzServer BMC, DDR SPD, PCIe slot sensors

We will implement a driver for the LM75 — the industry-standard I2C temperature sensor used on virtually every server motherboard. It reads temperature with 0.125°C resolution and has a programmable thermostat interrupt threshold.

Chapter 4 — Complete Temperature Monitor Driver

Driver Design

Our temp_monitor driver will:

  1. Register as an I2C client driver for the LM75 temperature sensor
  2. Expose temperature readings via Linux's hwmon subsystem (standard /sys/class/hwmon/hwmon0/temp1_input interface)
  3. Register a character device at /dev/temp_monitor for custom ioctl access
  4. Handle the LM75's interrupt (OS alert when temperature exceeds threshold)
  5. Use a kernel workqueue for deferred interrupt processing

Complete Source Code

/*
 * temp_monitor.c — Complete LM75 Temperature Sensor Driver
 *
 * Demonstrates: I2C client driver, hwmon integration, character device,
 * interrupt handling, workqueue, sysfs attributes, and proper error paths.
 *
 * Hardware: LM75 I2C temperature sensor (address 0x48–0x4F)
 * Datasheet: TI LM75A, NXP LM75B
 *
 * Build: obj-m += temp_monitor.o
 * Load:  sudo modprobe i2c-dev
 *        sudo insmod temp_monitor.ko
 *        echo "lm75 0x48" | sudo tee /sys/bus/i2c/devices/i2c-1/new_device
 */

#include <linux/module.h>
#include <linux/init.h>
#include <linux/i2c.h>
#include <linux/hwmon.h>
#include <linux/hwmon-sysfs.h>
#include <linux/interrupt.h>
#include <linux/workqueue.h>
#include <linux/mutex.h>
#include <linux/cdev.h>
#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/delay.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Linux Kernel for AI/ML");
MODULE_DESCRIPTION("LM75 Temperature Monitor with hwmon + char device");

/* ── LM75 register map ── */
#define LM75_REG_TEMP    0x00   /* 16-bit temperature (9-bit, 0.5°C LSB) */
#define LM75_REG_CONF    0x01   /* 8-bit config: shutdown, comparator/interrupt mode */
#define LM75_REG_THYST   0x02   /* hysteresis temperature */
#define LM75_REG_TOS     0x03   /* over-temperature shutdown threshold */

/* ── ioctl commands ── */
#define TEMP_IOC_MAGIC       'T'
#define TEMP_GET_CELSIUS     _IOR(TEMP_IOC_MAGIC, 0, __s32)
#define TEMP_SET_THRESHOLD   _IOW(TEMP_IOC_MAGIC, 1, __s32)
#define TEMP_GET_THRESHOLD   _IOR(TEMP_IOC_MAGIC, 2, __s32)

/* ── Driver private data — one instance per sensor ── */
struct temp_monitor_data {
    struct i2c_client  *client;       /* I2C device */
    struct device      *hwmon_dev;    /* hwmon device */
    struct cdev         cdev;        /* character device */
    struct mutex        lock;        /* protects sensor access */
    struct work_struct  alert_work;  /* workqueue for interrupt handling */
    s32                 last_temp_mc;/* last temperature in milli-Celsius */
    bool                alert_active;/* over-temperature alert pending */
    int                 threshold_c; /* alert threshold in Celsius */
};

static dev_t            temp_devno;
static struct class     *temp_class;

/* ══════════════════════════════════════════════════════
 * I2C TEMPERATURE READING
 * ══════════════════════════════════════════════════════ */

/* Read raw 16-bit register from LM75.
 * LM75 uses big-endian 16-bit values. */
static int lm75_read_reg16(struct i2c_client *client, u8 reg, s16 *val)
{
    u8 buf[2];
    int ret;

    /* I2C write: send register address */
    ret = i2c_master_send(client, &reg, 1);
    if (ret < 0) return ret;

    /* I2C read: receive 2 bytes */
    ret = i2c_master_recv(client, buf, 2);
    if (ret < 0) return ret;

    /* Convert big-endian: buf[0] = MSB, buf[1] = LSB */
    *val = (s16)(((u16)buf[0] << 8) | buf[1]);
    return 0;
}

/* Convert raw LM75 register value to milli-Celsius.
 * LM75: bits[15:7] = temperature, bit 6 = 0.5°C, bits[5:0] = 0
 * Resolution: 0.5°C per LSB (bit 6 of the 16-bit value) */
static s32 lm75_raw_to_mc(s16 raw)
{
    /* Right-shift 7 to get integer degrees (with sign extension) */
    s32 temp_c  = (s32)(raw >> 7);    /* integer Celsius */
    s32 temp_mc = temp_c * 1000;
    /* Add 0.5°C if bit 7 is set */
    if (raw & 0x80) temp_mc += 500;
    return temp_mc;
}

/* Read current temperature in milli-Celsius. Caller holds lock. */
static int temp_read_mc(struct temp_monitor_data *data, s32 *temp_mc)
{
    s16 raw;
    int ret = lm75_read_reg16(data->client, LM75_REG_TEMP, &raw);
    if (ret < 0) return ret;
    *temp_mc = lm75_raw_to_mc(raw);
    data->last_temp_mc = *temp_mc;
    return 0;
}

/* ══════════════════════════════════════════════════════
 * HWMON SUBSYSTEM INTEGRATION
 * Exposes: /sys/class/hwmon/hwmon0/temp1_input (milli-Celsius)
 *          /sys/class/hwmon/hwmon0/temp1_max   (threshold)
 * ══════════════════════════════════════════════════════ */

static umode_t temp_is_visible(const void *data,
                               enum hwmon_sensor_types type,
                               u32 attr, int channel)
{
    if (type == hwmon_temp) {
        switch (attr) {
        case hwmon_temp_input: return 0444;   /* read-only */
        case hwmon_temp_max:   return 0644;   /* read-write */
        }
    }
    return 0;
}

static int temp_hwmon_read(struct device *dev,
                            enum hwmon_sensor_types type,
                            u32 attr, int channel, long *val)
{
    struct temp_monitor_data *data = dev_get_drvdata(dev);
    s32 temp_mc;
    int ret;

    if (type != hwmon_temp) return -EOPNOTSUPP;

    switch (attr) {
    case hwmon_temp_input:
        mutex_lock(&data->lock);
        ret = temp_read_mc(data, &temp_mc);
        mutex_unlock(&data->lock);
        if (ret) return ret;
        *val = temp_mc;
        return 0;
    case hwmon_temp_max:
        *val = (long)data->threshold_c * 1000;
        return 0;
    }
    return -EOPNOTSUPP;
}

static int temp_hwmon_write(struct device *dev,
                             enum hwmon_sensor_types type,
                             u32 attr, int channel, long val)
{
    struct temp_monitor_data *data = dev_get_drvdata(dev);
    if (type == hwmon_temp && attr == hwmon_temp_max) {
        data->threshold_c = (int)(val / 1000);
        return 0;
    }
    return -EOPNOTSUPP;
}

static const struct hwmon_ops temp_hwmon_ops = {
    .is_visible = temp_is_visible,
    .read       = temp_hwmon_read,
    .write      = temp_hwmon_write,
};

static const u32 temp_chip_config[] = {
    HWMON_T_INPUT | HWMON_T_MAX,
    0
};
static const struct hwmon_channel_info temp_chip_info = {
    .type   = hwmon_temp,
    .config = temp_chip_config,
};
static const struct hwmon_channel_info *temp_info[] = {
    &temp_chip_info, NULL
};
static const struct hwmon_chip_info temp_chip = {
    .ops  = &temp_hwmon_ops,
    .info = temp_info,
};

/* ══════════════════════════════════════════════════════
 * CHARACTER DEVICE — /dev/temp_monitor
 * ══════════════════════════════════════════════════════ */

static int temp_open(struct inode *inode, struct file *file)
{
    struct temp_monitor_data *data =
        container_of(inode->i_cdev, struct temp_monitor_data, cdev);
    file->private_data = data;
    dev_dbg(&data->client->dev, "opened\n");
    return 0;
}

static ssize_t temp_read(struct file *file, char __user *buf,
                          size_t count, loff_t *pos)
{
    struct temp_monitor_data *data = file->private_data;
    s32 temp_mc;
    int ret;
    char kbuf[32];
    ssize_t len;

    mutex_lock(&data->lock);
    ret = temp_read_mc(data, &temp_mc);
    mutex_unlock(&data->lock);
    if (ret) return ret;

    /* Format: "23456\n" = 23.456 degrees Celsius */
    len = scnprintf(kbuf, sizeof(kbuf), "%d\n", temp_mc);
    if (count < len) return -EINVAL;
    if (copy_to_user(buf, kbuf, len)) return -EFAULT;
    return len;
}

static long temp_ioctl(struct file *file, unsigned int cmd,
                        unsigned long arg)
{
    struct temp_monitor_data *data = file->private_data;
    s32 temp_mc;
    int ret, threshold;

    switch (cmd) {
    case TEMP_GET_CELSIUS:
        mutex_lock(&data->lock);
        ret = temp_read_mc(data, &temp_mc);
        mutex_unlock(&data->lock);
        if (ret) return ret;
        return put_user(temp_mc, (__s32 __user *)arg);

    case TEMP_SET_THRESHOLD:
        if (get_user(threshold, (int __user *)arg)) return -EFAULT;
        if (threshold < -55 || threshold > 125) return -EINVAL;
        data->threshold_c = threshold;
        dev_info(&data->client->dev,
                 "Threshold set to %d°C\n", threshold);
        return 0;

    case TEMP_GET_THRESHOLD:
        return put_user(data->threshold_c, (__s32 __user *)arg);
    }
    return -ENOTTY;
}

static const struct file_operations temp_fops = {
    .owner          = THIS_MODULE,
    .open           = temp_open,
    .read           = temp_read,
    .unlocked_ioctl = temp_ioctl,
};

/* ══════════════════════════════════════════════════════
 * INTERRUPT HANDLER — triggered when T > threshold
 * Uses threaded IRQ so we can sleep in the handler
 * ══════════════════════════════════════════════════════ */

static irqreturn_t temp_alert_handler(int irq, void *dev_id)
{
    struct temp_monitor_data *data = dev_id;
    /* Defer actual work to process context via workqueue.
     * The IRQ handler itself must NOT sleep. */
    data->alert_active = true;
    schedule_work(&data->alert_work);
    return IRQ_HANDLED;
}

static void temp_alert_work_fn(struct work_struct *work)
{
    struct temp_monitor_data *data =
        container_of(work, struct temp_monitor_data, alert_work);
    s32 temp_mc;

    /* Now in process context — we CAN sleep if needed */
    mutex_lock(&data->lock);
    temp_read_mc(data, &temp_mc);
    mutex_unlock(&data->lock);

    dev_warn(&data->client->dev,
             "THERMAL ALERT! Temperature: %d.%03d°C (threshold: %d°C)\n",
             temp_mc / 1000, temp_mc % 1000, data->threshold_c);

    /* In a real driver: send kernel event, throttle workload,
     * initiate emergency shutdown if needed */
    data->alert_active = false;
}

/* ══════════════════════════════════════════════════════
 * I2C PROBE — called when the kernel matches our driver to hardware
 * ══════════════════════════════════════════════════════ */

static int temp_probe(struct i2c_client *client)
{
    struct temp_monitor_data *data;
    int ret;

    /* Allocate driver private data, zero-initialized, freed automatically
     * when the device is removed (managed resource via devm_*) */
    data = devm_kzalloc(&client->dev, sizeof(*data), GFP_KERNEL);
    if (!data) return -ENOMEM;

    data->client       = client;
    data->threshold_c  = 75;    /* 75°C default alert threshold */
    mutex_init(&data->lock);
    INIT_WORK(&data->alert_work, temp_alert_work_fn);
    i2c_set_clientdata(client, data);

    /* Verify sensor responds correctly */
    {
        s32 temp_mc;
        ret = temp_read_mc(data, &temp_mc);
        if (ret) {
            dev_err(&client->dev, "sensor not responding: %d\n", ret);
            return ret;
        }
        dev_info(&client->dev,
                 "LM75 found at 0x%02x, temp=%d.%03d°C\n",
                 client->addr, temp_mc / 1000, temp_mc % 1000);
    }

    /* Register with hwmon subsystem → /sys/class/hwmon/hwmonN/ */
    data->hwmon_dev = devm_hwmon_device_register_with_info(
        &client->dev, "temp_monitor", data, &temp_chip, NULL);
    if (IS_ERR(data->hwmon_dev))
        return PTR_ERR(data->hwmon_dev);

    /* Allocate a character device number */
    ret = alloc_chrdev_region(&temp_devno, 0, 1, "temp_monitor");
    if (ret) return ret;

    /* Create /dev/temp_monitor */
    cdev_init(&data->cdev, &temp_fops);
    data->cdev.owner = THIS_MODULE;
    ret = cdev_add(&data->cdev, temp_devno, 1);
    if (ret) { unregister_chrdev_region(temp_devno, 1); return ret; }

    temp_class = class_create(THIS_MODULE, "temp_monitor");
    device_create(temp_class, NULL, temp_devno, NULL, "temp_monitor");

    /* Register interrupt if IRQ is specified in device tree / ACPI */
    if (client->irq > 0) {
        ret = devm_request_threaded_irq(
            &client->dev, client->irq,
            NULL,                   /* hard IRQ handler (NULL = threaded only) */
            temp_alert_handler,     /* threaded handler */
            IRQF_TRIGGER_LOW | IRQF_ONESHOT,
            "temp_monitor_alert", data);
        if (ret)
            dev_warn(&client->dev, "IRQ %d unavailable: %d\n",
                     client->irq, ret);
        else
            dev_info(&client->dev, "Registered alert IRQ %d\n", client->irq);
    }

    dev_info(&client->dev,
             "temp_monitor ready: /dev/temp_monitor, "
             "/sys/class/hwmon/hwmon*/temp1_input\n");
    return 0;
}

static void temp_remove(struct i2c_client *client)
{
    struct temp_monitor_data *data = i2c_get_clientdata(client);
    cancel_work_sync(&data->alert_work);   /* wait for pending work */
    device_destroy(temp_class, temp_devno);
    class_destroy(temp_class);
    cdev_del(&data->cdev);
    unregister_chrdev_region(temp_devno, 1);
    dev_info(&client->dev, "temp_monitor unloaded\n");
}

    { "lm75a", 0 },

MODULE_DEVICE_TABLE(i2c, temp_id_table);

Python User-Space Client

#!/usr/bin/env python3
"""temp_monitor_client.py — Read temperature sensor from Python.
   Works with both the hwmon sysfs interface and our char device."""

import os, struct, fcntl, time
from pathlib import Path

# ── ioctl command numbers (must match driver constants) ──
TEMP_IOC_MAGIC     = ord('T')
TEMP_GET_CELSIUS   = (2 << 30) | (TEMP_IOC_MAGIC << 8) | 0 | (4 << 16)
TEMP_SET_THRESHOLD = (1 << 30) | (TEMP_IOC_MAGIC << 8) | 1 | (4 << 16)

class TempMonitor:
    def __init__(self, dev="/dev/temp_monitor"):
        self.dev = dev
        self.fd  = open(dev, "rb")

    def read_celsius(self) -> float:
        """Read temperature in Celsius via ioctl."""
        buf = struct.pack("i", 0)   # 4-byte output buffer
        result = fcntl.ioctl(self.fd, TEMP_GET_CELSIUS, buf)
        temp_mc = struct.unpack("i", result)[0]
        return temp_mc / 1000.0

    def read_sysfs(self) -> float:
        """Read temperature via hwmon sysfs (always works)."""
        hwmon_dirs = list(Path("/sys/class/hwmon").glob("hwmon*/temp1_input"))
        if not hwmon_dirs:
            raise RuntimeError("No hwmon temperature sensor found")
        val = int(hwmon_dirs[0].read_text())
        return val / 1000.0

    def close(self):
        self.fd.close()

def main():
    monitor = TempMonitor()
    print("Temperature Monitor — press Ctrl+C to stop\n")
    try:
        while True:
            temp_ioctl = monitor.read_celsius()
            temp_sysfs = monitor.read_sysfs()
            print(f"ioctl: {temp_ioctl:.3f}°C  |  sysfs: {temp_sysfs:.3f}°C",
                  end="\r")
            time.sleep(1.0)
    except KeyboardInterrupt:
        print("\nDone.")
    finally:
        monitor.close()

if __name__ == "__main__":
    main()

Chapter 5 — Interrupt Handling & DMA

The Two-Level Interrupt Model

Linux uses a two-level interrupt model for all hardware interrupts:

  1. Top half (hard IRQ handler): Runs immediately when the interrupt fires, with interrupts disabled on the current CPU. Must be blazing fast — no sleeping, no allocation. Acknowledges the hardware and schedules the bottom half.
  2. Bottom half (softirq / tasklet / workqueue): Runs in a deferred context. Can sleep (workqueues only). Does the actual work.
/* Two-level interrupt example: simulating a GPU "kernel complete" interrupt */
#include <linux/interrupt.h>
#include <linux/workqueue.h>

struct gpu_device {
    int             irq;
    struct work_struct  completion_work;
    volatile bool   kernel_done;
    u64             completion_status;
};

/* TOP HALF: runs with IRQs disabled, must complete in microseconds */
static irqreturn_t gpu_irq_handler(int irq, void *dev_id)
{
    struct gpu_device *gpu = dev_id;

    /* 1. Read and clear the interrupt status register (hardware-specific) */
    gpu->completion_status = readq(gpu_bar0 + GPU_IRQ_STATUS_REG);
    writeq(0xFFFFFFFF, gpu_bar0 + GPU_IRQ_CLEAR_REG);  /* ack interrupt */

    /* 2. Schedule bottom half for actual work */
    schedule_work(&gpu->completion_work);

    return IRQ_HANDLED;
}

/* BOTTOM HALF: runs in process context, can sleep, takes time */
static void gpu_completion_work(struct work_struct *work)
{

DMA for GPU Data Transfer — How cudaMemcpy Works

DMA (Direct Memory Access) allows the GPU to read and write host memory without CPU involvement. Understanding DMA is key to understanding why cudaMemcpyAsync() with pinned memory is so much faster than regular transfers.

/* DMA mapping for a GPU transfer buffer.
 * This is what the CUDA driver does internally for cudaMemcpyAsync. */
#include <linux/dma-mapping.h>

struct gpu_dma_buf {
    void        *vaddr;     /* CPU virtual address */
    dma_addr_t   dma_addr;  /* GPU/DMA physical address (IOVA) */
    size_t       size;
};

/* Allocate a DMA-coherent buffer.
 * "Coherent" = CPU and GPU always see the same data (no cache flush needed).
 * More expensive than streaming DMA but simpler to use. */
int gpu_alloc_dma_buffer(struct device *dev,
                          size_t size,
                          struct gpu_dma_buf *buf)
{
    buf->size  = size;
    buf->vaddr = dma_alloc_coherent(dev, size, &buf->dma_addr, GFP_KERNEL);
    if (!buf->vaddr) return -ENOMEM;

    dev_dbg(dev, "DMA buffer: vaddr=%p dma_addr=0x%llx size=%zu\n",
            buf->vaddr, (u64)buf->dma_addr, size);
    return 0;
}

void gpu_free_dma_buffer(struct device *dev, struct gpu_dma_buf *buf)
{
    dma_free_coherent(dev, buf->size, buf->vaddr, buf->dma_addr);
}

/* Program the GPU's DMA engine to transfer data.
 * In a real GPU driver, this writes to PCIe BAR registers that control
 * the GPU's internal DMA engine (copy engine). */
void gpu_program_dma_transfer(void __iomem *bar0,
                               dma_addr_t src, dma_addr_t dst,
                               size_t size)
{
    writeq(src,  bar0 + GPU_DMA_SRC_REG);
    writeq(dst,  bar0 + GPU_DMA_DST_REG);
    writeq(size, bar0 + GPU_DMA_SIZE_REG);
    writeq(1ULL, bar0 + GPU_DMA_START_REG);  /* fire! */
}

Chapter 6 — PCIe Driver Fundamentals

Every GPU is a PCIe device. The PCIe driver framework lets the kernel discover, initialise, and manage GPU cards. Understanding the PCIe probe sequence is essential for diagnosing "GPU not found" issues on new hardware.

#include <linux/pci.h>

/* PCI device ID table — the kernel matches these against lspci output */
static const struct pci_device_id myaccel_ids[] = {
    { PCI_DEVICE(0x1234, 0x5678) },   /* vendor_id, device_id */
    { }
};
MODULE_DEVICE_TABLE(pci, myaccel_ids);

static int myaccel_probe(struct pci_dev *pdev,
                          const struct pci_device_id *id)
{
    void __iomem *bar0;
    int ret;

    /* Enable the PCIe device (powers on, assigns resources) */
    ret = pci_enable_device(pdev);
    if (ret) return ret;

    /* Become the bus master — required for DMA */
    pci_set_master(pdev);

    /* Request exclusive access to BAR0 (the GPU's MMIO register space) */
    ret = pci_request_region(pdev, 0, "myaccel_bar0");
    if (ret) { pci_disable_device(pdev); return ret; }

    /* Map BAR0 into kernel virtual address space */
    bar0 = pci_iomap(pdev, 0, 0);   /* 0 = map full BAR */
    if (!bar0) { pci_release_region(pdev, 0); pci_disable_device(pdev); return -ENOMEM; }

    /* Set up 64-bit DMA addressing (modern GPUs always have 64-bit DMA) */
    ret = dma_set_mask_and_coherent(&pdev->dev, DMA_BIT_MASK(64));
    if (ret) dev_warn(&pdev->dev, "64-bit DMA not available\n");

    dev_info(&pdev->dev, "Accelerator at BAR0=%p, IRQ=%d\n", bar0, pdev->irq);

    /* Store bar0 in driver private data for later use */
    pci_set_drvdata(pdev, bar0);
    return 0;
}

static void myaccel_remove(struct pci_dev *pdev)
{
    void __iomem *bar0 = pci_get_drvdata(pdev);
    pci_iounmap(pdev, bar0);
    pci_release_region(pdev, 0);
    pci_disable_device(pdev);
}

static struct pci_driver myaccel_driver = {
    .name     = "myaccel",
    .id_table = myaccel_ids,
    .probe    = myaccel_probe,
    .remove   = myaccel_remove,
};
module_pci_driver(myaccel_driver);
