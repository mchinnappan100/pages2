From a minimal autoregressive loop to a full production-ready inference pipeline.

🔧 Minimal Autoregressive Inference

The simplest complete inference loop: tokenize the prompt, repeatedly call the model for the next-token logits, sample or argmax, append, repeat until EOS or max length.

pythonMinimal Autoregressive Generation (PyTorch)
import torch
import torch.nn.functional as F
from transformers import AutoTokenizer, AutoModelForCausalLM

def greedy_generate(
    model, tokenizer, prompt: str,
    max_new_tokens: int = 200,
    device: str = "cuda"
) -> str:
    """Pure autoregressive greedy decoding — no sampling."""
    model.eval()
    input_ids = tokenizer(prompt, return_tensors="pt").input_ids.to(device)
    generated  = input_ids.clone()

    with torch.no_grad():
        for _ in range(max_new_tokens):
            # Forward pass: (batch, seq_len, vocab_size)
            logits     = model(generated).logits
            next_logit = logits[:, -1, :]          # last token position
            next_token = next_logit.argmax(dim=-1, keepdim=True)

            # Append generated token
            generated = torch.cat([generated, next_token], dim=-1)

            # Stop on EOS token
            if next_token.item() == tokenizer.eos_token_id:
                break

    # Decode only the newly generated tokens
    new_tokens = generated[0, input_ids.shape[-1]:]
    return tokenizer.decode(new_tokens, skip_special_tokens=True)


# Usage
model_id  = "meta-llama/Meta-Llama-3-8B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_id)
model     = AutoModelForCausalLM.from_pretrained(
    model_id, torch_dtype=torch.bfloat16, device_map="auto"
)

prompt  = "Explain the KV cache in one paragraph:"
output  = greedy_generate(model, tokenizer, prompt)
print(output)

🏎️ Production Inference with KV Cache & Flash Attention

The production version enables Flash Attention 2, uses the model's built-in generate method (which includes KV caching automatically), and adds BF16 + torch.compile for maximum speed.

pythonProduction-Grade Inference (PyTorch)
import torch
from transformers import (AutoTokenizer, AutoModelForCausalLM,
                           TextStreamer, GenerationConfig)

class LlamaInference:
    def __init__(self, model_id: str, device: str = "cuda"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_id)
        self.tokenizer.pad_token = self.tokenizer.eos_token

        self.model = AutoModelForCausalLM.from_pretrained(
            model_id,
            torch_dtype=torch.bfloat16,
            device_map="auto",
            attn_implementation="flash_attention_2",   # FA2
        )
        self.model.eval()

        # Compile the model for faster repeated calls
        # self.model = torch.compile(self.model, mode="reduce-overhead")

        self.gen_config = GenerationConfig(
            do_sample=True,
            temperature=0.7,
            top_p=0.9,
            top_k=50,
            repetition_penalty=1.1,
            max_new_tokens=512,
            pad_token_id=self.tokenizer.eos_token_id,
            eos_token_id=self.tokenizer.eos_token_id,
        )

    def generate(self, prompt: str, stream: bool = False) -> str:
        # Format as chat template
        messages = [{"role": "user", "content": prompt}]
        text     = self.tokenizer.apply_chat_template(
            messages, tokenize=False, add_generation_prompt=True
        )
        inputs   = self.tokenizer(text, return_tensors="pt").to(self.model.device)

        streamer = TextStreamer(self.tokenizer, skip_prompt=True) if stream else None

        with torch.inference_mode():   # faster than no_grad()
            output_ids = self.model.generate(
                **inputs,
                generation_config=self.gen_config,
                streamer=streamer,
                use_cache=True,       # ← enables KV cache
            )

        new_ids = output_ids[0, inputs["input_ids"].shape[-1]:]
        return self.tokenizer.decode(new_ids, skip_special_tokens=True)

    def batch_generate(self, prompts: list[str]) -> list[str]:
        """Batched generation for higher throughput."""
        texts  = [self.tokenizer.apply_chat_template(
                      [{"role": "user", "content": p}],
                      tokenize=False, add_generation_prompt=True)
                  for p in prompts]
        inputs = self.tokenizer(texts, return_tensors="pt", padding=True,
                                truncation=True).to(self.model.device)
        with torch.inference_mode():
            out = self.model.generate(**inputs, generation_config=self.gen_config)
        results = []
        for i, ids in enumerate(out):
            new  = ids[inputs["input_ids"].shape[-1]:]
            results.append(self.tokenizer.decode(new, skip_special_tokens=True))
        return results


# Example usage
engine = LlamaInference("meta-llama/Meta-Llama-3-8B-Instruct")
print(engine.generate("What is Flash Attention?", stream=True))

Sampling Strategies & Decoding

Controlling the randomness and quality of generated text — the art and science of sampling.

🎲 Temperature Scaling

Temperature divides logits before softmax. Low temperature sharpens the distribution (more deterministic), high temperature flattens it (more random).

pythonComplete Sampling Strategies
import torch
import torch.nn.functional as F

def sample_next_token(
    logits: torch.Tensor,          # (vocab_size,)
    temperature: float = 1.0,
    top_k:       int   = 0,
    top_p:       float = 1.0,
    repetition_penalty: float = 1.0,
    past_ids:    torch.Tensor = None,
) -> int:
    """Apply temperature, top-k, top-p, and repetition penalty."""

    # 1. Repetition penalty: reduce probability of seen tokens
    if repetition_penalty != 1.0 and past_ids is not None:
        for token_id in past_ids.unique():
            if logits[token_id] > 0:
                logits[token_id] /= repetition_penalty
            else:
                logits[token_id] *= repetition_penalty

    # 2. Temperature scaling
    if temperature != 1.0:
        logits = logits / temperature

    # 3. Top-k filtering: keep only top k tokens
    if top_k > 0:
        threshold = torch.topk(logits, top_k).values[-1]
        logits    = logits.masked_fill(logits < threshold, float('-inf'))

    # 4. Top-p (nucleus) sampling: keep smallest set summing to ≥ p
    if top_p < 1.0:
        sorted_logits, sorted_idx = torch.sort(logits, descending=True)
        cum_probs = sorted_logits.softmax(dim=-1).cumsum(dim=-1)
        # Remove tokens with cumulative probability above threshold
        remove_mask = cum_probs - sorted_logits.softmax(dim=-1) > top_p
        sorted_logits[remove_mask] = float('-inf')
        # Scatter back to original order
        logits = torch.zeros_like(logits).scatter_(0, sorted_idx, sorted_logits)

    # 5. Sample from the final distribution
    probs = F.softmax(logits, dim=-1)
    return torch.multinomial(probs, num_samples=1).item()


# Greedy: temperature=0 (argmax)
# Creative writing: temp=1.0, top_p=0.95
# Code generation: temp=0.2, top_k=40
# Factual QA: temp=0.1 (near-deterministic)

⚡ Speculative Decoding

Speculative decoding uses a small fast draft model to propose K tokens, then verifies all K with the large model in a single forward pass. Accepted tokens are kept; the first rejected token is resampled from the target distribution.

pythonSpeculative Decoding
import torch
import torch.nn.functional as F

def speculative_decode(
    draft_model,   target_model,   tokenizer,
    prompt: str,   K: int = 4,    max_new: int = 200
) -> str:
    ids = tokenizer(prompt, return_tensors="pt").input_ids.cuda()
    total_accepted = 0

    while ids.shape[-1] - len(tokenizer(prompt).input_ids) < max_new:
        # ── Draft: autoregressively generate K candidate tokens ──
        draft_ids = ids.clone()
        draft_probs = []
        for _ in range(K):
            with torch.no_grad():
                dlogits = draft_model(draft_ids).logits[:, -1, :]
            dp = F.softmax(dlogits / 0.7, dim=-1)
            dt = torch.multinomial(dp, 1)
            draft_ids   = torch.cat([draft_ids, dt], dim=-1)
            draft_probs.append(dp[0, dt[0, 0]].item())

        # ── Target: verify all K+1 positions in ONE forward pass ──
        with torch.no_grad():
            tlogits = target_model(draft_ids).logits[0, -K-1:, :]
        target_probs = F.softmax(tlogits / 0.7, dim=-1)

        # ── Accept/reject each draft token ──
        n_accepted = 0
        for i in range(K):
            t_id    = draft_ids[0, ids.shape[-1] + i].item()
            q       = target_probs[i, t_id].item()
            p       = draft_probs[i]
            r       = torch.rand(1).item()
            if r <= min(1.0, q / p):
                n_accepted += 1
            else:
                break   # reject this and all subsequent draft tokens

        # Keep accepted tokens + one target-sampled correction token
        ids = draft_ids[:, :ids.shape[-1] + n_accepted]
        bonus = torch.multinomial(target_probs[n_accepted], 1).unsqueeze(0)
        ids   = torch.cat([ids, bonus], dim=-1)
        total_accepted += n_accepted + 1

        if ids[0, -1].item() == tokenizer.eos_token_id:
            break

    new_ids = ids[0, len(tokenizer(prompt).input_ids):]
    return tokenizer.decode(new_ids, skip_special_tokens=True)

Inference with JAX

Functional, JIT-compiled inference — ideal for TPU clusters and research workflows requiring extreme reproducibility.

🔑 JAX Inference Fundamentals

JAX's key advantages for inference: jax.jit compiles Python functions to XLA, eliminating Python overhead on repeated calls; jax.vmap vectorizes functions over batch dimensions; jax.pmap shards computation across devices. Unlike PyTorch, JAX uses explicit random keys for reproducibility.

pythonJAX Transformer Inference (Flax)
import jax
import jax.numpy as jnp
from flax import linen as nn
from functools import partial

class CausalSelfAttention(nn.Module):
    n_heads: int
    d_model: int

    @nn.compact
    def __call__(self, x, kv_cache=None, deterministic=True):
        B, T, C = x.shape
        H       = self.n_heads
        D       = C // H

        q = nn.Dense(C, use_bias=False, name="wq")(x).reshape(B, T, H, D)
        k = nn.Dense(C, use_bias=False, name="wk")(x).reshape(B, T, H, D)
        v = nn.Dense(C, use_bias=False, name="wv")(x).reshape(B, T, H, D)

        # Append to KV cache if provided (for autoregressive decode)
        if kv_cache is not None:
            k_cache, v_cache = kv_cache
            k = jnp.concatenate([k_cache, k], axis=1)
            v = jnp.concatenate([v_cache, v], axis=1)
        new_cache = (k, v)

        # Transpose for batched matmul: (B, H, T, D)
        q, k, v = [a.transpose(0, 2, 1, 3) for a in (q, k, v)]

        scale  = D ** -0.5
        scores = jnp.matmul(q, k.swapaxes(-1, -2)) * scale

        # Causal mask
        Tq, Tk = scores.shape[-2], scores.shape[-1]
        mask   = jnp.tril(jnp.ones((Tq, Tk), dtype=bool), k=Tk - Tq)
        scores = jnp.where(mask, scores, jnp.finfo(scores.dtype).min)

        attn   = jax.nn.softmax(scores, axis=-1)
        out    = jnp.matmul(attn, v).transpose(0, 2, 1, 3).reshape(B, T, C)
        out    = nn.Dense(C, use_bias=False, name="wo")(out)
        return out, new_cache


class TransformerBlock(nn.Module):
    n_heads: int
    d_model: int
    d_ffn:   int

    @nn.compact
    def __call__(self, x, kv_cache=None):
        # Pre-norm + attention
        x_norm = nn.RMSNorm()(x)
        attn_out, new_cache = CausalSelfAttention(
            self.n_heads, self.d_model)(x_norm, kv_cache)
        x = x + attn_out

        # Pre-norm + FFN (SwiGLU)
        x_norm = nn.RMSNorm()(x)
        gate   = nn.Dense(self.d_ffn, use_bias=False)(x_norm)
        up     = nn.Dense(self.d_ffn, use_bias=False)(x_norm)
        ffn    = nn.Dense(self.d_model, use_bias=False)(jax.nn.silu(gate) * up)
        x = x + ffn
        return x, new_cache

⚡ JIT-Compiled Autoregressive Generation in JAX

pythonJAX JIT Inference Loop
import jax
import jax.numpy as jnp
from jax import random

@partial(jax.jit, static_argnames=["temperature", "top_k"])
def generate_step(params, input_ids, kv_cache, rng_key,
                   temperature: float = 0.7, top_k: int = 50):
    """Single JAX-JIT-compiled generation step with KV cache."""
    # Model forward pass
    logits, new_cache = model.apply(params, input_ids, kv_cache=kv_cache)
    next_logits = logits[:, -1, :] / temperature   # (B, vocab)

    # Top-k filtering
    top_vals, _ = jax.lax.top_k(next_logits, top_k)
    threshold   = top_vals[:, -1:] - 1e-6
    next_logits = jnp.where(next_logits >= threshold, next_logits, -1e9)

    # Sample — JAX requires explicit RNG key (no global state)
    next_token = random.categorical(rng_key, next_logits)  # (B,)
    return next_token, new_cache


def jax_generate(params, prompt_ids: jnp.ndarray,
                  max_new_tokens: int = 200, seed: int = 42):
    """Full JAX autoregressive generation loop."""
    rng        = random.PRNGKey(seed)
    ids        = prompt_ids           # (1, prompt_len)
    kv_cache   = None                 # initialised on first step
    generated  = []

    for _ in range(max_new_tokens):
        rng, step_rng = random.split(rng)
        next_tok, kv_cache = generate_step(
            params, ids, kv_cache, step_rng
        )
        generated.append(int(next_tok[0]))
        ids = next_tok[:, jnp.newaxis]   # only feed new token (KV cache holds past)
        if int(next_tok[0]) == EOS_ID:
            break

    return generated


# On first call: JIT compilation (~10-30 sec), subsequent calls: native speed
# Benefit: Python overhead eliminated entirely after first call

🌐 Multi-Device Inference with pmap

pythonJAX Multi-GPU Inference with pmap
import jax
from jax import pmap
import jax.numpy as jnp

n_devices = jax.device_count()
print(f"Available devices: {n_devices}")  # e.g., 4 GPUs or 8 TPU cores

# Replicate model params across all devices
replicated_params = jax.device_put_replicated(params, jax.devices())

@pmap
def parallel_forward(params, input_ids):
    """Run inference on all devices simultaneously — different batch shards."""
    logits, _ = model.apply(params, input_ids)
    return logits[:, -1, :]   # (local_batch, vocab)

# Batch must be divisible by n_devices
batch_size = 16
input_ids  = jnp.ones((16, 512), dtype=jnp.int32)

# Reshape for pmap: (n_devices, batch_per_device, seq_len)
sharded_ids = input_ids.reshape(n_devices, batch_size // n_devices, 512)
logits      = parallel_forward(replicated_params, sharded_ids)
# logits shape: (n_devices, batch_per_device, vocab_size)
