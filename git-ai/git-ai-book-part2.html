11.1 The Error System

Before any command runs, every error must carry a typed exit code. This is the foundation all commands build on.

commands/errors.go — typed exit codesgo
package commands

import "fmt"

// exitErr pairs an error message with a numeric exit code.
// Exit codes per spec §15:
//   0 — success          3 — provider unavailable
//   1 — generic error    4 — model unavailable
//   2 — repo not found   5 — invalid config
type exitErr struct {
    code int
    msg  string
}

func (e *exitErr) Error() string  { return e.msg }
func (e *exitErr) ExitCode() int  { return e.code }

// exitError constructs a typed error. main.go reads ExitCode() for os.Exit.
func exitError(code int, format string, a ...any) error {
    return &exitErr{code: code, msg: fmt.Sprintf(format, a...)}
}

11.2 The Main Entry Point

cmd/git-ai/main.gogo
package main

func main() {
    root := &cobra.Command{
        Use:           "git-ai",
        Short:         "AI-powered Git assistant (local LLM via Ollama)",
        Version:       "0.2.0",
        SilenceUsage:  true,   // don't print usage on error
        SilenceErrors: true,   // we format errors ourselves
        // Forward unknown subcommands to git-ai-<name> plugin binaries.
        RunE: func(cmd *cobra.Command, args []string) error {
            if len(args) == 0 { return cmd.Help() }
            return plugin.Run(args[0], args[1:])
        },
    }

    root.AddCommand(
        commands.NewCommitCmd(),
        commands.NewReviewCmd(),
        commands.NewExplainCmd(),
        commands.NewDiffCmd(),
        commands.NewChangelogCmd(),
        commands.NewReleaseCmd(),
        commands.NewConflictsCmd(),
        commands.NewRepoCmd(),
        commands.NewAskCmd(),
        commands.NewPluginsCmd(),
    )

    if err := root.Execute(); err != nil {
        fmt.Fprintf(os.Stderr, "git-ai: %s\n", err)
        code := 1
        var ec exitCoder
        if errors.As(err, &ec) { code = ec.ExitCode() }
        os.Exit(code)
    }
}

11.3 git ai commit — Full Command

commands/commit.go — key sections annotatedgo
func NewCommitCmd() *cobra.Command {
    var style string   // "conventional" | "short" | "detailed"
    var amend bool    // true = regenerate message for HEAD

    cmd := &cobra.Command{
        Use:   "commit",
        Short: "Generate a commit message for staged changes",
        RunE: func(cmd *cobra.Command, args []string) error {
            fmtFlag := resolveFormat(addFormatFlagValue(cmd))
            cfg, _ := config.Load()
            meta, err := git.Metadata()
            if err != nil { return exitError(2, "not a git repository") }

            var diff string
            if amend {
                // --amend: show HEAD diff + any new staged changes
                diff, _ = git.CommitShow("HEAD")
                if staged, _ := git.StagedDiff(); staged != "" {
                    diff += "\n\n--- Additional staged changes ---\n" + staged
                }
                existing, _ := git.Run("log", "-1", "--pretty=%B")
                meta["existing_message"] = strings.TrimSpace(existing)
            } else {
                diff, err = git.StagedDiff()
                if err != nil || diff == "" {
                    return exitError(1, "no staged changes — run `git add` first")
                }
            }

            p := ollama.New(cfg.Ollama.URL, cfg.Model)
            if ok, err := p.IsAvailable(); !ok {
                return exitError(3, "Ollama not available: %v", err)
            }

            kind := "commit"
            if amend { kind = "commit_amend" }
            prpt := prompt.Build(kind, meta, diff, cfg.MaxTokens)
            result, err := p.Generate(context.Background(), prpt)

            if fmtFlag == output.FormatJSON {
                output.PrintJSON(map[string]string{
                    "message": strings.TrimSpace(result),
                    "amend":   fmt.Sprintf("%v", amend),
                })
                return nil
            }
            output.Header("Suggested commit message", fmtFlag)
            fmt.Println(result)
            output.Advisory(fmtFlag)
            return nil
        },
    }
    cmd.Flags().StringVar(&style, "style", "", "conventional, short, detailed")
    cmd.Flags().BoolVar(&amend, "amend", false, "regenerate message for HEAD")
    addFormatFlag(cmd)
    return cmd
}

11.4 git ai explain — Concept vs. Ref Detection

explain has a unique challenge: its argument can be a commit ref (HEAD, abc1234) or a Git concept word (rebase, stash). The detection is a simple map lookup:

commands/explain.go — concept detectiongo
// gitConcepts is the set of argument values treated as Git concepts, not refs.
// Anything not in this map goes to CommitShow() as a ref.
var gitConcepts = map[string]bool{
    "rebase": true, "cherry-pick": true, "stash": true,
    "bisect": true, "reflog":      true, "merge": true,
    "reset":  true, "revert":      true, "tag":   true,
}

if gitConcepts[target] {
    // Concept — no git repo required, explain the term.
    meta := map[string]string{"concept": target}
    prpt = prompt.Build("explain_concept", meta, "", cfg.MaxTokens)
} else {
    // Ref — retrieve the commit diff and explain it.
    diff, err = git.CommitShow(target)
    prpt = prompt.Build("explain_commit", meta, diff, cfg.MaxTokens)
}

// If --from is not provided, find the last tag.
// If no tags exist, use the very first commit (entire project history).
if from == "" {
    lastTag, err := git.Run("describe", "--tags", "--abbrev=0")
    if err != nil || lastTag == "" {
        // No tags — use the root commit so the full history is covered.
        firstCommit, err := git.Run("rev-list", "--max-parents=0", "HEAD")
        if err != nil { return exitError(1, "could not determine range") }
        from = firstCommit
    } else {
        from = lastTag
    }
}

The "say so explicitly" rule
The ask prompt instructs: "If the answer cannot be determined from the context, say so explicitly." Without this, models confidently answer questions the context doesn't support. With it, the model says "I cannot determine this from the available context" — which is the honest and correct answer.

The Plugin System

Extending git-ai without touching its source — the same model Git itself uses

14.1 How It Works

git-ai extends Git's own extensibility model one level deeper. Just as git ai maps to git-ai, a plugin binary git-ai-security becomes reachable as git ai security. This requires zero changes to git-ai itself.

git ai security --severity high │ ▼ main.go: unknown command "security" │ ▼ (cobra RunE fallback) plugin.Run("security", ["--severity", "high"]) │ ▼ exec.LookPath("git-ai-security") ← searches $PATH │ ▼ exec.Command("git-ai-security", "--severity", "high") │ stdin/stdout/stderr inherited ▼ plugin executes, output flows to terminal

14.2 Full Implementation

internal/plugin/plugin.gogo
// Run finds and executes a git-ai-<name> binary.
// Stdin/stdout/stderr are inherited so the plugin is indistinguishable
// from a built-in command from the user's perspective.
func Run(name string, args []string) error {
    binary := "git-ai-" + name

    // LookPath uses the same PATH search algorithm as the shell.
    path, err := exec.LookPath(binary)
    if err != nil {
        return fmt.Errorf("plugin %q not found (looked for %q on PATH)", name, binary)
    }

    cmd := exec.Command(path, args...)
    cmd.Stdin  = os.Stdin    // allow interactive plugins
    cmd.Stdout = os.Stdout   // output flows to terminal
    cmd.Stderr = os.Stderr   // errors visible immediately
    cmd.Env    = os.Environ() // full environment (PATH, HOME, etc.)
    return cmd.Run()
}

// List returns all git-ai-* binaries found on PATH.
// Used by `git ai plugins` to show available extensions.
func List() []string {
    dirs := strings.Split(os.Getenv("PATH"), string(os.PathListSeparator))
    seen := map[string]bool{}
    var plugins []string
    for _, dir := range dirs {
        entries, _ := os.ReadDir(dir)
        for _, e := range entries {
            name := e.Name()
            if strings.HasPrefix(name, "git-ai-") && !e.IsDir() {
                plug := strings.TrimPrefix(name, "git-ai-")
                if !seen[plug] { seen[plug] = true; plugins = append(plugins, plug) }
            }
        }
    }
    return plugins
}

Output Formatting

Three formats, one flag, zero decorative noise in JSON mode

15.1 The Format-Aware Output Functions

internal/output/output.go — format-aware functionsgo
// Header prints a section header — completely silent in JSON mode.
// A header line in JSON output would corrupt the payload.
func Header(label string, format Format) {
    if format == FormatJSON { return }
    fmt.Printf("\n%s\n%s\n", label, strings.Repeat("-", len(label)))
}

// RepoInfo prints branch/HEAD/provider/model — silent in JSON mode.
func RepoInfo(branch, head, provider, model string, format Format) {
    if format == FormatJSON { return }
    fmt.Printf("  Branch:   %s\n  HEAD:     %s\n  Provider: %s\n  Model:    %s\n\n",
        branch, head, provider, model)
}

// Advisory prints "No changes made" footer — silent in JSON mode.
func Advisory(format Format) {
    if format == FormatJSON { return }
    fmt.Println("\nNo changes have been made to your repository.")
}

// PrintJSON emits an arbitrary value as indented JSON — the only output
// function called in JSON mode. All other decorative output is suppressed.
func PrintJSON(v any) {
    b, _ := json.MarshalIndent(v, "", "  ")
    fmt.Println(string(b))
}

Building with Make

Every Makefile target explained, cross-platform builds, goreleaser

16.1 The Makefile

Makefile — annotated line by linemakefile
# .PHONY declares targets that don't produce files.
# Without this, make would skip a target if a file named "build" existed.
.PHONY: build install test lint clean

# Output binary path — referenced in build and install targets.
BINARY := bin/git-ai

# build: compile the binary.
# -ldflags="-s -w"
#   -s  strips the symbol table      (~15% size reduction)
#   -w  strips DWARF debug info       (~15% size reduction)
# Combined: ~8 MB instead of ~12 MB, with zero runtime impact.
build:
	go build -ldflags="-s -w" -o $(BINARY) ./cmd/git-ai

# install: build then copy to /usr/local/bin.
# After this, `git ai` works in any terminal, any directory.
install: build
	cp $(BINARY) /usr/local/bin/git-ai
	@echo "Installed git-ai to /usr/local/bin/git-ai"

# test: run all *_test.go files in the module.
# ./... is Go's recursive package glob.
test:
	go test ./...

# lint: static analysis (requires: brew install golangci-lint).
lint:
	golangci-lint run

# clean: remove compiled artifacts.
clean:
	rm -rf bin/

Chapter 17

Wiring to Git

The exact mechanism by which git ai finds and runs the binary

17.1 How Git Discovers External Commands

Git's command dispatch works in two steps. This is documented in git help git:

User types: git ai commit Step 1: Git checks built-in commands commit ✓, push ✓, pull ✓, log ✓ ... "ai" ✗ — not a built-in Step 2: Git searches PATH for "git-ai" /usr/local/bin/git-ai ← found! Step 3: Git executes: /usr/local/bin/git-ai commit Result: cobra receives ["commit"] as os.Args[1:] → NewCommitCmd() runs
The rule
git <subcommand> → Git looks for git-<subcommand> on PATH. The binary name must match exactly. The binary must be executable. That is the entire contract.

// mockServer creates a real HTTP server that responds like Ollama.
func mockServer(t *testing.T, response string) *httptest.Server {
    t.Helper()
    return httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        switch r.URL.Path {
        case "/api/generate":
            json.NewEncoder(w).Encode(generateResponse{Response: response})
        case "/api/tags":
            json.NewEncoder(w).Encode(map[string]any{
                "models": []map[string]string{
                    {"name": "llama3"}, {"name": "mistral"},
                },
            })
        default:
            w.WriteHeader(http.StatusOK) // health check
        }
    }))
}

func TestGenerate(t *testing.T) {
    srv := mockServer(t, "feat: add feature")
    defer srv.Close()

    p := New(srv.URL, "llama3")
    result, err := p.Generate(context.Background(), "write a commit message")
    if err != nil { t.Fatalf("Generate() error: %v", err) }
    if result != "feat: add feature" {
        t.Errorf("unexpected result: %q", result)
    }
}

func TestIsAvailableUnreachable(t *testing.T) {
    // Port 1 is reserved and never has anything listening.
    p := New("http://127.0.0.1:1", "llama3")
    ok, err := p.IsAvailable()
    if ok  { t.Error("expected available=false") }
    if err == nil { t.Error("expected non-nil error") }
}

19.3 Architecture Lessons

🔌

Lesson 1: Define the interface first

  • Provider interface before first implementation
  • Adding OpenAI/llama.cpp = zero command changes
  • Interface is the right abstraction boundary
📝

Lesson 2: Prompts as code

  • Version-controlled Go functions
  • Testable, type-checked, deterministic
  • Bad prompt = compile-time or test-time failure
🔧

Lesson 3: JSON output from day one

  • Makes every command scriptable
  • Turns git-ai into a platform
  • CI pipelines, Slack bots, IDE plugins
The invariant that made everything else possible
Committing to "AI never modifies the repository" at the design phase made every other decision simpler. Commands don't need undo logic. Output doesn't need confirmation prompts. The safety guarantee is structural, not behavioral — it cannot be accidentally violated.
