Arrays, Devices & PRNG

JAX arrays are immutable; randomness is explicit via splittable keys.

arrays & devices
import jax
import jax.numpy as jnp

# list available devices
print(jax.devices())         # [CpuDevice(id=0)] or [GpuDevice(id=0), ...]

x = jnp.ones((4, 8, 512))   # (B, T, D)  — on default device
x = jax.device_put(x, jax.devices("gpu")[0])

# dtype control
x = jnp.zeros((3, 3), dtype=jnp.bfloat16)
x = x.astype(jnp.float32)

# JAX arrays are immutable — no in-place ops
# y = x.at[0].set(1.0)      # functional "mutation"
PRNG — explicit, splittable keys
key = jax.random.PRNGKey(42)          # seed once

# NEVER reuse a key — always split first
key, subkey = jax.random.split(key)
x = jax.random.normal(subkey, (4, 512))

# split into N keys at once
keys = jax.random.split(key, num=8)   # (8,) array of keys

# common distributions
jax.random.uniform(subkey, (3, 3))
jax.random.randint(subkey, (B, T), 0, vocab_size)
jax.random.bernoulli(subkey, p=0.1, shape=(D,))
shape ops you'll use constantly
x.reshape(B, T, H, D)           # returns new array
jnp.transpose(x, (0, 2, 1, 3)) # reorder dims
jnp.swapaxes(x, 1, 2)           # swap two dims
jnp.expand_dims(x, axis=0)      # add dim  -> (1, ...)
jnp.squeeze(x, axis=-1)         # drop size-1 dim
jnp.concatenate([a, b], axis=-1)# concat
jnp.stack([a, b], axis=0)       # new dim

# functional "mutation"
x = x.at[0, :].set(0.0)        # returns modified copy
x = x.at[mask].add(1.0)
pytrees — JAX's universal container
# Any nested dict/list/tuple of arrays is a pytree
params = {"w": jnp.ones((D, D)), "b": jnp.zeros(D)}

# map a fn over every leaf
scaled = jax.tree.map(lambda p: p * 0.02, params)

# flatten / unflatten
leaves, treedef = jax.tree.flatten(params)
restored = treedef.unflatten(leaves)

# used everywhere: model params, optimizer state, batches
Gotcha — JAX arrays are immutable. Operations like x[0] = 1 silently fail or error. Use x.at[0].set(1) which returns a new array. Under jit, .at[] ops compile to in-place XLA scatter ops with no copy overhead.

The Core Transforms: jit · grad · vmap · pmap

JAX is a function-transformation library. These four transforms compose freely.

jax.jit — compile & cache
@jax.jit
def forward(params, x):
    return x @ params["w"] + params["b"]

# first call: traces and compiles (slow)
# subsequent calls: runs compiled XLA binary (fast)
out = forward(params, x)

# static args that must NOT be traced (shapes, Python ints)
@functools.partial(jax.jit, static_argnames=("training",))
def apply(params, x, training=False):
    ...

# donate argnums: donate buffer ownership to XLA (saves memory)
@functools.partial(jax.jit, donate_argnums=(0,))
def train_step(state, batch): ...
jax.grad — automatic differentiation
def loss_fn(params, x, y):
    logits = forward(params, x)
    return cross_entropy(logits, y)

# grad of scalar loss w.r.t. params (argnums=0 default)
grad_fn = jax.grad(loss_fn)
grads = grad_fn(params, x, y)

# value AND grad in one pass
loss, grads = jax.value_and_grad(loss_fn)(params, x, y)

# grad w.r.t. multiple args
grad_fn = jax.grad(loss_fn, argnums=(0, 1))
g_params, g_x = grad_fn(params, x, y)

# jit-compile the backward pass too
grad_fn = jax.jit(jax.value_and_grad(loss_fn))
jax.vmap — auto-batching
# write for a single example, vmap adds the batch dim
def single_forward(params, x):    # x: (T, D)
    return x @ params["w"]        # (T, D)

# batched version: x: (B, T, D) -> (B, T, D)
batched = jax.vmap(single_forward, in_axes=(None, 0))
out = batched(params, x_batch)

# in_axes=None means "broadcast this arg, don't batch it"
# useful for per-sample gradients (no loops needed):
per_sample_grads = jax.vmap(
    jax.grad(loss_fn), in_axes=(None, 0, 0)
)(params, xs, ys)
jax.pmap — SPMD across devices
n_devices = jax.device_count()

# replicate params across all devices
params_rep = jax.tree.map(
    lambda p: jnp.stack([p] * n_devices), params
)

# shard batch: first dim split across devices
x_sharded = x.reshape(n_devices, -1, T, D)

@jax.pmap
def train_step(params, batch):
    loss, grads = jax.value_and_grad(loss_fn)(params, batch)
    # all-reduce grads across devices
    grads = jax.lax.pmean(grads, axis_name="batch")
    return loss, grads

loss, grads = train_step(params_rep, x_sharded)
Composability — transforms nest freely: jax.jit(jax.vmap(jax.grad(f))) just works. The compiler sees the full fused computation and optimizes it as one XLA program. This is the key advantage over PyTorch's eager-then-compile approach.
No Python side effects under jit — print(), list appends, and global mutations are only executed during tracing (first call). Use jax.debug.print("{x}", x=x) for in-graph printing that fires every call.

Transformer Building Blocks in Flax NNX

Flax NNX (the modern API) uses stateful Python objects, but params stay in JAX-compatible pytrees.

multi-head attention
from flax import nnx
import jax.numpy as jnp

class MultiHeadAttention(nnx.Module):
    def __init__(self, d_model: int, n_heads: int, rngs: nnx.Rngs):
        assert d_model % n_heads == 0
        self.n_heads = n_heads
        self.d_head  = d_model // n_heads
        self.qkv  = nnx.Linear(d_model, 3 * d_model, use_bias=False, rngs=rngs)
        self.proj = nnx.Linear(d_model, d_model,     use_bias=False, rngs=rngs)

    def __call__(self, x, mask=None):
        B, T, D = x.shape
        q, k, v = jnp.split(self.qkv(x), 3, axis=-1)   # each (B,T,D)

        def split_heads(t):
            return t.reshape(B, T, self.n_heads, self.d_head).transpose(0, 2, 1, 3)

        q, k, v = map(split_heads, (q, k, v))            # (B, H, T, d_head)

        scale = self.d_head ** -0.5
        scores = jnp.einsum("bhid,bhjd->bhij", q, k) * scale  # (B,H,T,T)
        if mask is not None:
            scores = scores + mask
        weights = jax.nn.softmax(scores, axis=-1)
        out = jnp.einsum("bhij,bhjd->bhid", weights, v)        # (B,H,T,d_head)
        out = out.transpose(0, 2, 1, 3).reshape(B, T, D)
        return self.proj(out)
Tip: swap the manual einsum attention above for jax.nn.dot_product_attention(q, k, v, mask=mask) (JAX 0.4.25+) to get a Flash-Attention–style fused kernel on supported hardware.
RMSNorm
class RMSNorm(nnx.Module):
    def __init__(self, dim: int):
        self.scale = nnx.Param(jnp.ones(dim))

    def __call__(self, x):
        rms = jnp.sqrt(jnp.mean(x ** 2, axis=-1, keepdims=True) + 1e-6)
        return x / rms * self.scale.value
SwiGLU feed-forward
class SwiGLU(nnx.Module):
    def __init__(self, d_model: int, mult: int = 4, rngs: nnx.Rngs = None):
        hidden = int(2 * d_model * mult / 3)
        self.w1 = nnx.Linear(d_model, hidden, use_bias=False, rngs=rngs)
        self.w2 = nnx.Linear(d_model, hidden, use_bias=False, rngs=rngs)
        self.w3 = nnx.Linear(hidden, d_model,  use_bias=False, rngs=rngs)

    def __call__(self, x):
        return self.w3(jax.nn.silu(self.w1(x)) * self.w2(x))
RoPE positional embeddings
def rope_freqs(seq_len, d_head, base=10000):
    inv_freq = 1.0 / (base ** (
        jnp.arange(0, d_head, 2) / d_head))
    t = jnp.arange(seq_len)
    freqs = jnp.outer(t, inv_freq)          # (T, d_head/2)
    return jnp.concatenate([freqs, freqs], axis=-1)  # (T, d_head)

def apply_rope(x, freqs):
    x1, x2 = jnp.split(x, 2, axis=-1)
    rot = jnp.concatenate([-x2, x1], axis=-1)
    return x * jnp.cos(freqs) + rot * jnp.sin(freqs)
pre-norm transformer block
class TransformerBlock(nnx.Module):
    def __init__(self, d_model, n_heads, rngs):
        self.norm1 = RMSNorm(d_model)
        self.attn  = MultiHeadAttention(d_model, n_heads, rngs)
        self.norm2 = RMSNorm(d_model)
        self.ffn   = SwiGLU(d_model, rngs=rngs)

    def __call__(self, x, mask=None):
        x = x + self.attn(self.norm1(x), mask=mask)
        x = x + self.ffn(self.norm2(x))
        return x
Flax NNX vs Linen — NNX (the current recommended API) uses stateful Python objects like PyTorch. The older Linen API used functional module.init(key, x) / module.apply(params, x). New projects should use NNX; Linen is still common in existing codebases.

A Full GPT-Style Decoder in Flax NNX

Stack the block from Sheet 03 and add embeddings + an LM head.

class GPT(nnx.Module):
    def __init__(self, vocab_size, d_model=512, n_layers=12,
                 n_heads=8, max_seq=2048, rngs: nnx.Rngs = None):
        self.tok_emb = nnx.Embed(vocab_size, d_model, rngs=rngs)
        self.blocks  = [TransformerBlock(d_model, n_heads, rngs) for _ in range(n_layers)]
        self.norm_f  = RMSNorm(d_model)
        self.head    = nnx.Linear(d_model, vocab_size, use_bias=False, rngs=rngs)
        self.max_seq = max_seq
        # weight tying: share embedding and output projection weights
        self.head.kernel = self.tok_emb.embedding   # nnx.Variable reference

    def __call__(self, idx, training=False):
        B, T = idx.shape
        x = self.tok_emb(idx)                       # (B, T, D)

        # causal mask — upper-triangular filled with -inf
        mask = jnp.triu(jnp.full((T, T), -1e9), k=1)
        mask = mask[None, None, :, :]               # (1, 1, T, T)

        for block in self.blocks:
            x = block(x, mask=mask)

        x = self.norm_f(x)
        logits = self.head(x)                       # (B, T, vocab)
        return logits

    def loss(self, idx, targets):
        logits = self(idx)
        return cross_entropy_loss(logits, targets)

def cross_entropy_loss(logits, targets, ignore_index=-100):
    B, T, V = logits.shape
    logits_2d = logits.reshape(B * T, V)
    targets_1d = targets.reshape(B * T)
    log_probs = jax.nn.log_softmax(logits_2d, axis=-1)
    nll = -log_probs[jnp.arange(B * T), targets_1d]
    mask = targets_1d != ignore_index
    return jnp.sum(nll * mask) / jnp.sum(mask)

# instantiate
rngs = nnx.Rngs(params=jax.random.PRNGKey(0))
model = GPT(vocab_size=32000, rngs=rngs)

# count parameters
n_params = sum(x.size for x in jax.tree.leaves(nnx.state(model)))
print(f"{n_params / 1e6:.1f}M parameters")
nnx.state / nnx.graphdef

nnx.state(model) extracts a pure pytree of params. nnx.graphdef(model) captures the structure. Together they replace Linen's init/apply.

weight tying

Assign a nnx.Variable reference across modules. NNX de-duplicates shared variables in the state pytree automatically.

ignore_index=-100

Same convention as PyTorch — set prompt/padding targets to -100 and mask them out of the loss.

jit-compiled train step
@jax.jit
def train_step(state, batch):
    x, y = batch["input_ids"], batch["labels"]

    def loss_fn(params):
        logits = state.apply_fn(params, x, training=True)
        return cross_entropy_loss(logits, y)

    loss, grads = jax.value_and_grad(loss_fn)(state.params)
    state = state.apply_gradients(grads=grads)   # one-liner update
    return state, loss

# training loop
for step, batch in enumerate(dataloader):

        print(f"step {step}: loss={loss:.4f}, lr={schedule(state.step):.2e}")

jax.checkpoint (remat) recomputes activations on the backward pass instead of storing them — same trade-off as PyTorch's gradient checkpointing.

Stale closures under jit — jax.jit traces over Python control flow once. If a Python variable captured in the closure changes, the old compiled version keeps running. Put all changing state in function arguments, not closures.

causal mask
def causal_mask(T):
    # returns (1, 1, T, T) — broadcastable to (B, H, T, T)
    mask = jnp.triu(jnp.full((T, T), -1e9), k=1)
    return mask[None, None, :, :]

# for sliding window / chunked attention:
# mask out positions > window_size steps apart
def sliding_mask(T, window):
    idx = jnp.arange(T)
    diff = idx[:, None] - idx[None, :]       # (T, T)
    return jnp.where((diff < 0) | (diff > window), -1e9, 0.0)
padding mask for batched sequences
# pad_mask: (B, T) bool, True = real token
def make_attn_mask(pad_mask, causal=True):
    B, T = pad_mask.shape
    # key padding: (B, 1, 1, T) → block attention to pad tokens
    key_mask = jnp.where(pad_mask[:, None, None, :], 0.0, -1e9)
    if causal:
        key_mask = key_mask + causal_mask(T)
    return key_mask

Inference & Autoregressive Generation

inference

In JAX, the generation loop lives outside jit; each step calls a jit-compiled decode_step.

generation loop with static KV cache
def init_kv_cache(n_layers, B, max_seq, n_heads, d_head):
    shape = (B, n_heads, max_seq, d_head)
    return [{"k": jnp.zeros(shape), "v": jnp.zeros(shape)}
            for _ in range(n_layers)]

@jax.jit
def decode_step(params, token, pos, kv_cache):
    # single-token forward; updates cache in-place (functional .at)
    x = embed(params, token)                           # (B, 1, D)
    new_cache = []
    for i, (block_p, cache) in enumerate(zip(params["blocks"], kv_cache)):
        x, updated = attention_with_cache(block_p, x, cache, pos)
        new_cache.append(updated)
        x = ffn(block_p, x)
    logits = lm_head(params, x[:, 0, :])              # (B, vocab)
    return logits, new_cache

def generate(params, prompt_ids, max_new=200, temperature=1.0, top_k=50, key=None):
    B, T = prompt_ids.shape
    kv_cache = init_kv_cache(n_layers, B, T + max_new, n_heads, d_head)

    # prefill: process whole prompt
    for pos in range(T - 1):
        _, kv_cache = decode_step(params, prompt_ids[:, pos:pos+1], pos, kv_cache)

    tokens = [prompt_ids[:, -1:]]
    for pos in range(T - 1, T + max_new - 1):
        logits, kv_cache = decode_step(params, tokens[-1], pos, kv_cache)
        logits = logits / temperature

        if top_k is not None:
            topk_vals = jnp.sort(logits, axis=-1)[:, -top_k:-top_k+1]
            logits = jnp.where(logits < topk_vals, -1e9, logits)

        key, subkey = jax.random.split(key)
        next_tok = jax.random.categorical(subkey, logits)[:, None]
        tokens.append(next_tok)

    return jnp.concatenate(tokens, axis=1)

Static (pre-allocated) cache avoids recompilation on every step — jit compiles once for the fixed shape and replays the XLA binary each decode step.

sampling strategy cheat sheet
temperature → 0greedy (argmax), deterministic
temperature ~0.7-1.0balanced for chat/completion
top_k=40-100hard cutoff to k most likely
top_p=0.9-0.95nucleus — smallest set covering p mass
greedy decode (fastest)
next_tok = jnp.argmax(logits, axis=-1, keepdims=True)
# deterministic — use for evals / reproducibility

# top-p (nucleus) sampling
def top_p_sample(logits, p, key):
    probs = jax.nn.softmax(logits, axis=-1)
    sorted_probs = jnp.sort(probs, axis=-1)[:, ::-1]
    cumprobs = jnp.cumsum(sorted_probs, axis=-1)
    cutoff = jnp.where(cumprobs - sorted_probs > p,
                        0.0, sorted_probs).min(axis=-1, keepdims=True)
    logits = jnp.where(probs < cutoff, -1e9, logits)
    return jax.random.categorical(key, logits)
Recompilation danger — any change to a jit-compiled function's array shapes triggers a full recompile. Keep your KV cache pre-allocated at max_seq and use a position index, not growing arrays, to avoid shape changes mid-generation.

Distributed Training & Sharding

training

JAX exposes devices as a mesh; shard arrays across it with PartitionSpec.

device mesh & PartitionSpec
from jax.sharding import Mesh, PartitionSpec as P
from jax.sharding import NamedSharding
import numpy as np

devices = np.array(jax.devices()).reshape(2, 4)  # 2 nodes × 4 GPUs

with Mesh(devices, ("dp", "tp")) as mesh:
    # data-parallel: batch dim sharded over "dp"
    x_sharding = NamedSharding(mesh, P("dp", None, None))
    # tensor-parallel: weight dim sharded over "tp"
    w_sharding = NamedSharding(mesh, P(None, "tp"))

    x = jax.device_put(x, x_sharding)
jit with in/out shardings
@functools.partial(
    jax.jit,
    in_shardings=(replicated_sharding, data_sharding),
    out_shardings=replicated_sharding,
)
def train_step(params, batch):
    loss, grads = jax.value_and_grad(loss_fn)(params, batch)
    # grads are automatically all-reduced across "dp" axis
    params = jax.tree.map(lambda p, g: p - lr * g, params, grads)
    return params, loss
pmap — simpler data parallelism
n = jax.device_count()

# replicate params across all devices
params_rep = jax.tree.map(lambda p: jnp.stack([p] * n), params)
# shard batch
x_sharded = x.reshape(n, -1, T, D)

@functools.partial(jax.pmap, axis_name="batch")
def train_step(params, x, y):
    loss, grads = jax.value_and_grad(loss_fn)(params, x, y)
    grads = jax.lax.pmean(grads, axis_name="batch")
    return loss, grads

losses, grads = train_step(params_rep, x_sharded, y_sharded)
mean_loss = losses[0]   # identical on all replicas after pmean

pmap is simpler than mesh sharding; use it for pure data parallelism. For model/tensor parallelism, prefer the Mesh approach above.

memory levers, ranked by ease
  1. 1. bfloat16 activations
  2. 2. gradient accumulation (optax.MultiSteps)
  3. 3. jax.checkpoint (remat) on blocks
  4. 4. lax.scan over layers (O(1) activation stack)
  5. 5. tensor parallelism via Mesh + PartitionSpec
