nn.Module skeleton
class Block(nn.Module):
    def __init__(self, d_model):
        super().__init__()
        self.fc = nn.Linear(d_model, d_model)
        self.act = nn.GELU()

    def forward(self, x):
        return self.act(self.fc(x))

m = Block(512).to(device)
m.train()   # enables dropout / batchnorm updates
m.eval()    # disables them — always call before inference
Gotcha — view() fails silently-loud (a hard error) on non-contiguous tensors after a transpose/permute. Call .contiguous() first, or just use .reshape().
02

The Transformer Building Blocks

Embeddings → positions → attention → feed-forward → norm. In that order, repeated.

scaled dot-product attention

Attention(Q,K,V) = softmax(QKᵀ / √d_k · mask) · V — every variant (MHA, GQA, cross-attn) is this call with different Q/K/V shapes. 

import torch.nn.functional as F

class MultiHeadAttention(nn.Module):
    def __init__(self, d_model, n_heads, dropout=0.1):
        super().__init__()
        assert d_model % n_heads == 0
        self.n_heads, self.d_head = n_heads, d_model // n_heads
        self.qkv  = nn.Linear(d_model, 3 * d_model, bias=False)
        self.proj = nn.Linear(d_model, d_model, bias=False)
        self.drop = dropout

    def forward(self, x, attn_mask=None, kv_cache=None):
        B, T, D = x.shape
        q, k, v = self.qkv(x).chunk(3, dim=-1)                       # each (B,T,D)
        q = q.view(B, T, self.n_heads, self.d_head).transpose(1, 2)  # (B,H,T,d)
        k = k.view(B, T, self.n_heads, self.d_head).transpose(1, 2)
        v = v.view(B, T, self.n_heads, self.d_head).transpose(1, 2)

        if kv_cache is not None:                    # inference: append & reuse
            k = torch.cat([kv_cache["k"], k], dim=2)
            v = torch.cat([kv_cache["v"], v], dim=2)
            kv_cache["k"], kv_cache["v"] = k, v

        # fused, memory-efficient kernel (flash-attention under the hood)
        out = F.scaled_dot_product_attention(
            q, k, v, attn_mask=attn_mask,
            dropout_p=self.drop if self.training else 0.0,
            is_causal=attn_mask is None,             # causal shortcut
        )                                            # (B,H,T,d)
        out = out.transpose(1, 2).contiguous().view(B, T, D)
        return self.proj(out)
Use F.scaled_dot_product_attention instead of hand-rolled softmax(QKᵀ) — it auto-dispatches to Flash-Attention / memory-efficient / math kernels and is the single biggest free speedup in modern PyTorch.
rotary positional embeddings (RoPE)
def rope_cache(seq_len, d_head, base=10000, device="cpu"):
    inv_freq = 1.0 / (base ** (torch.arange(0, d_head, 2, device=device) / d_head))
    t = torch.arange(seq_len, device=device)
    freqs = torch.outer(t, inv_freq)                # (T, d_head/2)
    return torch.cat([freqs, freqs], dim=-1)         # (T, d_head)

def apply_rope(x, freqs):
    x1, x2 = x.chunk(2, dim=-1)
    rot = torch.cat([-x2, x1], dim=-1)
    return x * freqs.cos() + rot * freqs.sin()
RMSNorm (LLaMA-style, cheaper than LayerNorm)
class RMSNorm(nn.Module):
    def __init__(self, dim, eps=1e-6):
        super().__init__()
        self.weight = nn.Parameter(torch.ones(dim))
        self.eps = eps

    def forward(self, x):
        rms = x.pow(2).mean(-1, keepdim=True).add(self.eps).rsqrt()
        return x * rms * self.weight
SwiGLU feed-forward
class SwiGLU(nn.Module):
    def __init__(self, d_model, mult=4):
        super().__init__()
        hidden = int(2 * d_model * mult / 3)
        self.w1 = nn.Linear(d_model, hidden, bias=False)
        self.w2 = nn.Linear(d_model, hidden, bias=False)
        self.w3 = nn.Linear(hidden, d_model, bias=False)

    def forward(self, x):
        return self.w3(F.silu(self.w1(x)) * self.w2(x))
pre-norm transformer block
class TransformerBlock(nn.Module):
    def __init__(self, d_model, n_heads):
        super().__init__()
        self.norm1 = RMSNorm(d_model)
        self.attn  = MultiHeadAttention(d_model, n_heads)
        self.norm2 = RMSNorm(d_model)
        self.ffn   = SwiGLU(d_model)

    def forward(self, x, **kw):
        x = x + self.attn(self.norm1(x), **kw)   # residual #1
        x = x + self.ffn(self.norm2(x))           # residual #2
        return x
Pre-norm vs post-norm — pre-norm (norm inside the residual branch, as above) trains far more stably at depth > ~12 layers without a warmup crutch. Nearly every modern LLM (GPT-2 onward) uses pre-norm.
03

A Full GPT-Style Decoder

Stack the block from Sheet 02 and add embeddings + an LM head. This is the whole model.

class GPT(nn.Module):
    def __init__(self, vocab_size, d_model=512, n_layers=12, n_heads=8, max_seq=2048):
        super().__init__()
        self.tok_emb = nn.Embedding(vocab_size, d_model)
        self.blocks  = nn.ModuleList([TransformerBlock(d_model, n_heads) for _ in range(n_layers)])
        self.norm_f  = RMSNorm(d_model)
        self.head    = nn.Linear(d_model, vocab_size, bias=False)
        self.head.weight = self.tok_emb.weight        # weight tying
        self.max_seq = max_seq
        self.apply(self._init_weights)

    def _init_weights(self, m):
        if isinstance(m, nn.Linear):
            nn.init.normal_(m.weight, mean=0.0, std=0.02)
        elif isinstance(m, nn.Embedding):
            nn.init.normal_(m.weight, mean=0.0, std=0.02)

    def forward(self, idx, targets=None):
        x = self.tok_emb(idx)                           # (B,T,D) — RoPE applied inside attn
        for block in self.blocks:
            x = block(x)
        x = self.norm_f(x)
        logits = self.head(x)                            # (B,T,vocab)

        loss = None
        if targets is not None:
            loss = F.cross_entropy(
                logits.view(-1, logits.size(-1)), targets.view(-1),
                ignore_index=-100,
            )
        return logits, loss

    def num_params(self):
        return sum(p.numel() for p in self.parameters())
weight tying

Sharing tok_emb.weight with the output head cuts params ~10–20% and is standard since GPT-2.

init std = 0.02

GPT-2's convention; scale residual-branch output projections by 1/√(2·n_layers) for deeper stability.

ignore_index=-100

PyTorch's cross-entropy default — mask padding/prompt tokens by setting their target to -100.

The Training Loop

training

AMP, gradient accumulation, clipping, and a cosine schedule with warmup — the modern default recipe.

production-grade loop
model = GPT(vocab_size=32000).to(device)
opt = torch.optim.AdamW(model.parameters(), lr=3e-4, betas=(0.9, 0.95),
                         weight_decay=0.1, fused=True)

scaler = torch.cuda.amp.GradScaler(enabled=(dtype == torch.float16))
accum_steps = 4
grad_clip = 1.0

model.train()
for step, (x, y) in enumerate(loader):
    x, y = x.to(device, non_blocking=True), y.to(device, non_blocking=True)

    with torch.autocast(device_type="cuda", dtype=torch.bfloat16):
        logits, loss = model(x, y)
        loss = loss / accum_steps

    scaler.scale(loss).backward()

    if (step + 1) % accum_steps == 0:
        scaler.unscale_(opt)
        torch.nn.utils.clip_grad_norm_(model.parameters(), grad_clip)
        scaler.step(opt)
        scaler.update()
        opt.zero_grad(set_to_none=True)
        scheduler.step()
cosine schedule with warmup
import math

def lr_lambda(step, warmup=2000, total=100_000, min_ratio=0.1):
    if step < warmup:
        return step / max(1, warmup)
    progress = (step - warmup) / max(1, total - warmup)
    cos = 0.5 * (1 + math.cos(math.pi * progress))
    return min_ratio + (1 - min_ratio) * cos

scheduler = torch.optim.lr_scheduler.LambdaLR(opt, lr_lambda)
checkpointing
torch.save({
    "model": model.state_dict(),
    "optimizer": opt.state_dict(),
    "scheduler": scheduler.state_dict(),
    "step": step,
}, f"ckpt_{step:07d}.pt")

ckpt = torch.load("ckpt_0100000.pt", map_location=device)
model.load_state_dict(ckpt["model"])
opt.load_state_dict(ckpt["optimizer"])
gradient checkpointing (activation recompute)
from torch.utils.checkpoint import checkpoint

class TransformerBlock(nn.Module):
    ...
    def forward(self, x, use_ckpt=True, **kw):
        if use_ckpt and self.training:
            return checkpoint(self._forward, x, use_reentrant=False)
        return self._forward(x, **kw)

Trades ~30% more compute for a large activation-memory reduction — lets you fit bigger batches / longer context.

# first step is slow (tracing); subsequent steps ~1.3-2x faster

def causal_mask(T, device):
    m = torch.full((T, T), float("-inf"), device=device)

Inference & Autoregressive Generation

inference

KV-caching turns O(T²) generation into O(T) per step. Everything else here is sampling strategy.

generation loop with KV cache
@torch.no_grad()
def generate(model, idx, max_new_tokens, temperature=1.0, top_k=None, top_p=None):
    model.eval()
    kv_caches = [{"k": None, "v": None} for _ in model.blocks]   # per-layer cache

    for _ in range(max_new_tokens):
        idx_cond = idx if kv_caches[0]["k"] is None else idx[:, -1:]  # 1 new tok after prefill
        logits, _ = model(idx_cond, kv_caches=kv_caches)
        logits = logits[:, -1, :] / temperature                       # (B, vocab)

        if top_k is not None:
            v, _ = torch.topk(logits, top_k)
            logits[logits < v[:, [-1]]] = float("-inf")

        if top_p is not None:
            sorted_logits, sorted_idx = torch.sort(logits, descending=True)
            probs = F.softmax(sorted_logits, dim=-1)
            cum = torch.cumsum(probs, dim=-1)
            mask = cum - probs > top_p
            sorted_logits[mask] = float("-inf")
            logits.scatter_(1, sorted_idx, sorted_logits)

        probs = F.softmax(logits, dim=-1)
        next_tok = torch.multinomial(probs, num_samples=1)             # (B,1)
        idx = torch.cat([idx, next_tok], dim=1)

        if (next_tok == eos_id).all():
            break
    return idx

First call ("prefill") processes the whole prompt at once and fills the cache; every call after sends only the newest token.

sampling strategy cheat sheet
temperature → 0greedy, deterministic, repetitive
temperature ~0.7-1.0balanced default for chat/completion
top_k=40-100hard cutoff to k most likely tokens
top_p=0.9-0.95nucleus — smallest set covering p mass
greedy decode (fastest, deterministic)
next_tok = logits.argmax(dim=-1, keepdim=True)
# no sampling, no temperature — use for evals / reproducibility
static KV cache (fixed-size, compile-friendly)
k_cache = torch.zeros(B, n_heads, max_seq, d_head, device=device)
v_cache = torch.zeros_like(k_cache)

def update_cache(k_cache, v_cache, k, v, pos):
    k_cache[:, :, pos:pos + k.size(2)] = k
    v_cache[:, :, pos:pos + v.size(2)] = v
    return k_cache[:, :, :pos + k.size(2)], v_cache[:, :, :pos + v.size(2)]

Pre-allocated caches avoid reallocation/concat every step and are required for torch.compile'd generation.

batched generation with left-padding
tokenizer.padding_side = "left"     # so new tokens append cleanly on the right
batch = tokenizer(prompts, padding=True, return_tensors="pt").to(device)
out = model.generate(**batch, max_new_tokens=200,
                      pad_token_id=tokenizer.pad_token_id)
Always model.eval() + torch.no_grad() for inference — forgetting either wastes memory building an autograd graph you'll never call .backward() on, and dropout left on will make outputs non-deterministic.
