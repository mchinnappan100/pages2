Tensors

01

What is a Tensor?

A tensor is a multi-dimensional array — the fundamental data structure in PyTorch. Similar to NumPy ndarrays but can run on GPU and supports automatic differentiation.

0-D: scalar, 1-D: vector, 2-D: matrix, N-D: tensor
Attributes: shape, dtype, device
Can share memory with NumPy arrays

Key dtypes

PyTorch supports multiple numeric types. Choosing the right dtype affects memory, speed, and precision.

torch.float32 — default, 32-bit float
torch.float16 / bfloat16 — mixed precision
torch.int64 — indices/labels
torch.bool — masks

Broadcasting Rules

PyTorch broadcasts tensors of different shapes by expanding dimensions of size 1. Essential for efficient vectorised operations.

Align shapes from the right
Size-1 dims expand to match
(3,1) + (1,4) → (3,4)

Autograd — Automatic Differentiation

📐 Basic Gradient Computation

autograd_basic.pyPython
# requires_grad=True tells PyTorch to track operations
x = torch.tensor(3.0, requires_grad=True)
y = x ** 2 + 2 * x + 1   # y = x² + 2x + 1
y.backward()              # compute dy/dx
print(x.grad)              # tensor(8.) — dy/dx = 2x+2 at x=3

# Multivariable gradient
x = torch.tensor([1., 2., 3.], requires_grad=True)
z = (x ** 2).sum()         # z = Σ xᵢ²
z.backward()
print(x.grad)              # tensor([2., 4., 6.]) — dz/dxᵢ = 2xᵢ

# grad_fn shows the operation that created the tensor
a = torch.randn(3, requires_grad=True)
b = a * 2
print(b.grad_fn)            # <MulBackward0>
print(b.requires_grad)     # True — inherits from a

🛑 Disabling Gradient Tracking

no_grad.pyPython
# Option 1: context manager (preferred for inference)
with torch.no_grad():
    out = model(x)           # no grad tracked, faster

# Option 2: inference_mode (even faster, stricter)
with torch.inference_mode():
    out = model(x)

# Option 3: detach from computation graph
x = torch.randn(3, requires_grad=True)
y = x * 2
y_detached = y.detach()    # new tensor, no grad_fn
print(y_detached.requires_grad)  # False

# Option 4: disable on parameter directly
for param in model.parameters():
    param.requires_grad = False   # freeze layer

🔄 Gradient Accumulation & Advanced Patterns

grad_advanced.pyPython
# Gradient accumulation (simulate large batch)
accum_steps = 4
optimizer.zero_grad()
for i, (x, y) in enumerate(loader):
    loss = criterion(model(x), y) / accum_steps
    loss.backward()

        optimizer.zero_grad()

# Custom autograd function
class MyReLU(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x):
        ctx.save_for_backward(x)
        return x.clamp(min=0)
    @staticmethod
    def backward(ctx, grad_output):
        x, = ctx.saved_tensors
        return grad_output * (x > 0).float()

x = torch.randn(4, requires_grad=True)
y = MyReLU.apply(x)

import torch.nn as nn

class MLP(nn.Module):
    def __init__(self, in_features, hidden, out_features):
        super().__init__()
        self.fc1    = nn.Linear(in_features, hidden)
        self.relu   = nn.ReLU()
        self.dropout= nn.Dropout(0.3)
        self.fc2    = nn.Linear(hidden, out_features)
        self.bn     = nn.BatchNorm1d(hidden)

    def forward(self, x):
        x = self.fc1(x)
        x = self.bn(x)
        x = self.relu(x)
        x = self.dropout(x)
        return self.fc2(x)

model = MLP(784, 256, 10)
print(model)
print(sum(p.numel() for p in model.parameters()))  # param count

class SimpleCNN(nn.Module):
    def __init__(self, num_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 32, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2),
            nn.Conv2d(32, 64, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2),
            nn.Conv2d(64, 128, 3, padding=1), nn.ReLU(),
        )
        self.pool = nn.AdaptiveAvgPool2d((1, 1))
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128, 256), nn.ReLU(), nn.Dropout(0.5),
            nn.Linear(256, num_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = self.pool(x)
        return self.classifier(x)

model = SimpleCNN()
x = torch.randn(4, 3, 32, 32)  # batch=4, RGB, 32×32

🔁 Standard Train / Validation Loop

training_loop.pyPython
def train_epoch(model, loader, criterion, optimizer, device):
    model.train()
    total_loss, correct, total = 0, 0, 0
    for x, y in loader:
        x, y = x.to(device), y.to(device)
        optimizer.zero_grad()           # 1. zero gradients
        out  = model(x)                  # 2. forward pass
        loss = criterion(out, y)         # 3. compute loss
        loss.backward()                 # 4. backward pass
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)  # 5. clip
        optimizer.step()                # 6. update weights
        total_loss += loss.item() * x.size(0)
        correct    += (out.argmax(1) == y).sum().item()
        total      += x.size(0)
    return total_loss/total, correct/total

@torch.no_grad()
def eval_epoch(model, loader, criterion, device):
    model.eval()
    total_loss, correct, total = 0, 0, 0
    for x, y in loader:
        x, y = x.to(device), y.to(device)
        out  = model(x)
        loss = criterion(out, y)
        total_loss += loss.item() * x.size(0)
        correct    += (out.argmax(1) == y).sum().item()
        total      += x.size(0)
    return total_loss/total, correct/total

# Main loop with early stopping
best_val_loss, patience, patience_counter = float('inf'), 10, 0
for epoch in range(100):
    tr_loss, tr_acc = train_epoch(model, train_loader, criterion, optimizer, device)
    vl_loss, vl_acc = eval_epoch(model, val_loader, criterion, device)
    scheduler.step(vl_loss)
    if vl_loss < best_val_loss:
        best_val_loss = vl_loss
        torch.save(model.state_dict(), 'best.pt')
        patience_counter = 0
    else:
        patience_counter += 1
        if patience_counter >= patience:
            print("Early stopping"); break
    print(f"Epoch {epoch}: train={tr_loss:.4f}/{tr_acc:.3f}  val={vl_loss:.4f}/{vl_acc:.3f}")

⚡ Mixed Precision Training (AMP)

amp_training.pyPython
from torch.cuda.amp import autocast, GradScaler

scaler = GradScaler()  # scales loss to prevent fp16 underflow

def train_amp(model, loader, criterion, optimizer, device):
    model.train()
    for x, y in loader:
        x, y = x.to(device), y.to(device)
        optimizer.zero_grad()
        with autocast(device_type='cuda', dtype=torch.float16):
            out  = model(x)          # fp16 forward
            loss = criterion(out, y) # fp16 loss
        scaler.scale(loss).backward()          # scaled backward
        scaler.unscale_(optimizer)             # unscale for clipping
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        scaler.step(optimizer)                 # step if grads are finite
        scaler.update()                        # update scale factor

# Sequential — ordered layers, single forward call
net = nn.Sequential(

# ModuleList — list of modules (manual forward logic)
class DeepNet(nn.Module):
    def __init__(self, sizes):
        super().__init__()
        self.layers = nn.ModuleList([
            nn.Linear(sizes[i], sizes[i+1]) for i in range(len(sizes)-1)
        ])
    def forward(self, x):
        for layer in self.layers[:-1]:
            x = torch.relu(layer(x))
        return self.layers[-1](x)

# ModuleDict — named modules (task-specific heads)
self.heads = nn.ModuleDict({
    'cls': nn.Linear(512, 10),
    'reg': nn.Linear(512, 1),
})

# Forward hook — inspect activations
activations = {}
def get_activation(name):
    def hook(module, input, output):
        activations[name] = output.detach()
    return hook

handle = model.fc1.register_forward_hook(get_activation('fc1'))

handle.remove()     # always remove hooks when done

# Backward hook — inspect gradients
grads = {}
def save_grad(name):
    def hook(grad):
        grads[name] = grad
    return hook
x.register_hook(save_grad('input'))

🏆 End-to-End CIFAR-10 Training Script

cifar10_complete.pyPython
import torch, torch.nn as nn
from torchvision import datasets, transforms, models
from torch.cuda.amp import autocast, GradScaler

device = 'cuda' if torch.cuda.is_available() else 'cpu'

# Data
tfm = transforms.Compose([
    transforms.RandomCrop(32, padding=4),
    transforms.RandomHorizontalFlip(),
    transforms.ToTensor(),
    transforms.Normalize((0.5,0.5,0.5), (0.5,0.5,0.5)),
])
train_ds = datasets.CIFAR10('data', train=True,  download=True, transform=tfm)
val_ds   = datasets.CIFAR10('data', train=False, download=True,
                              transform=transforms.Compose([transforms.ToTensor(),
                              transforms.Normalize((0.5,0.5,0.5),(0.5,0.5,0.5))]))
train_loader = torch.utils.data.DataLoader(train_ds, 128, shuffle=True,  num_workers=4, pin_memory=True)
val_loader   = torch.utils.data.DataLoader(val_ds,   256, shuffle=False, num_workers=4, pin_memory=True)

# Model (pretrained ResNet18, fine-tune head)
model = models.resnet18(weights=None)
model.fc = nn.Linear(model.fc.in_features, 10)
model = torch.compile(model.to(device))

criterion = nn.CrossEntropyLoss(label_smoothing=0.1)
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=0.05)
scheduler = torch.optim.lr_scheduler.OneCycleLR(optimizer, 0.01, epochs=30, steps_per_epoch=len(train_loader))
scaler    = GradScaler()

best_acc = 0
for epoch in range(30):
    model.train()
    for x, y in train_loader:
        x, y = x.to(device, non_blocking=True), y.to(device, non_blocking=True)
        optimizer.zero_grad()
        with autocast(device_type='cuda'):
            loss = criterion(model(x), y)
        scaler.scale(loss).backward()
        scaler.unscale_(optimizer)
        nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        scaler.step(optimizer); scaler.update()
        scheduler.step()
    model.eval()
    correct = 0
    with torch.no_grad():
        for x, y in val_loader:
            x, y = x.to(device), y.to(device)
            correct += (model(x).argmax(1) == y).sum().item()
    acc = correct / len(val_ds)
    if acc > best_acc:
        best_acc = acc
        torch.save(model.state_dict(), 'cifar10_best.pt')
    print(f"Epoch {epoch+1:02d}  val_acc={acc:.3f}  best={best_acc:.3f}")
